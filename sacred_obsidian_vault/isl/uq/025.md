 <h1 align="CENTER">25. ‏الفرقان‎</h1>
 <h2 align="CENTER">'lfrq'n</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_001">1</a> ‏تَبَارَكَ ٱلَّذِى نَزَّلَ ٱلْفُرْقَانَ عَلَىٰ عَبْدِهِۦ لِيَكُونَ لِلْعَلَمِينَ نَذِيرًا ‎</p>
 </td>
 <td width="50%">taba'raka 'l:aðiy naz:ala 'lfurqa'na ʕalay' ʕabdihiy liyakuwna lilʕalamiyna naðiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_002">2</a> ‏ٱلَّذِى لَهُۥ مُلْكُ ٱلسَّمَوَتِ وَٱلْأَرْضِ وَلَمْ يَتَّخِذْ وَلَدًۭا وَلَمْ يَكُن لَّهُۥ شَرِيكٌۭ فِى ٱلْمُلْكِ وَخَلَقَ كُلَّ شَىْءٍۢ فَقَدَّرَهُۥ تَقْدِيرًۭا ‎</p>
 </td>
 <td width="50%">'l:aðiy lahuw mulku 'ls:amawati wa'l'ʔarḍi walam yat:axið waladanm' walam yakun l:ahuw ʃariykunm fiy 'lmulki waxalaqa kul:a ʃayʔinm faqad:arahuw taqdiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_003">3</a> ‏وَٱتَّخَذُوا۟ مِن دُونِهِۦٓ ءَالِهَةًۭ لَّا يَخْلُقُونَ شَيْـًۭٔا وَهُمْ يُخْلَقُونَ وَلَا يَمْلِكُونَ لِأَنفُسِهِمْ ضَرًّۭا وَلَا نَفْعًۭا وَلَا يَمْلِكُونَ مَوْتًۭا وَلَا حَيَوٰةًۭ وَلَا نُشُورًۭا ‎</p>
 </td>
 <td width="50%">wa't:axaðuw' min duwnihiyʔ ʔa'lihatanm l:a' yaxluquwna ʃayʔanm' wahum yuxlaquwna wala' yamlikuwna li'ʔanfusihim ḍar:anm' wala' nafʕanm' wala' yamlikuwna mawtanm' wala' ḥayaw'tanm wala' nuʃuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_004">4</a> ‏وَقَالَ ٱلَّذِينَ كَفَرُوٓا۟ إِنْ هَذَآ إِلَّآ إِفْكٌ ٱفْتَرَىٰهُ وَأَعَانَهُۥ عَلَيْهِ قَوْمٌ ءَاخَرُونَ ۖ فَقَدْ جَآءُو ظُلْمًۭا وَزُورًۭا ‎</p>
 </td>
 <td width="50%">waqa'la 'l:aðiyna kafaruwʔ' 'ʔin haða'ʔ 'ʔil:a'ʔ 'ʔifkun 'ftaray'hu wa'ʔaʕa'nahuw ʕalayhi qawmun ʔa'xaruwna sl' faqad ja'ʔʔuw ẓulmanm' wazuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_005">5</a> ‏وَقَالُوٓا۟ أَسَطِيرُ ٱلْأَوَّلِينَ ٱكْتَتَبَهَا فَهِىَ تُمْلَىٰ عَلَيْهِ بُكْرَةًۭ وَأَصِيلًۭا ‎</p>
 </td>
 <td width="50%">waqa'luwʔ' 'ʔasaṭiyru 'l'ʔaw:aliyna 'ktatabaha' fahiya tumlay' ʕalayhi bukratanm wa'ʔaṣiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_006">6</a> ‏قُلْ أَنزَلَهُ ٱلَّذِى يَعْلَمُ ٱلسِّرَّ فِى ٱلسَّمَوَتِ وَٱلْأَرْضِ ۚ إِنَّهُۥ كَانَ غَفُورًۭا رَّحِيمًۭا ‎</p>
 </td>
 <td width="50%">qul 'ʔanzalahu 'l:aðiy yaʕlamu 'ls:ir:a fiy 'ls:amawati wa'l'ʔarḍi j 'ʔin:ahuw ka'na ɣafuwranm' r:aḥiymanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_007">7</a> ‏وَقَالُوا۟ مَالِ هَذَا ٱلرَّسُولِ يَأْكُلُ ٱلطَّعَامَ وَيَمْشِى فِى ٱلْأَسْوَاقِ ۙ لَوْلَآ أُنزِلَ إِلَيْهِ مَلَكٌۭ فَيَكُونَ مَعَهُۥ نَذِيرًا ‎</p>
 </td>
 <td width="50%">waqa'luw' ma'li haða' 'lr:asuwli ya'ʔkulu 'lṭ:aʕa'ma wayamʃiy fiy 'l'ʔaswa'qi l' lawla'ʔ 'ʔunzila 'ʔilayhi malakunm fayakuwna maʕahuw naðiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_008">8</a> ‏أَوْ يُلْقَىٰٓ إِلَيْهِ كَنزٌ أَوْ تَكُونُ لَهُۥ جَنَّةٌۭ يَأْكُلُ مِنْهَا ۚ وَقَالَ ٱلظَّلِمُونَ إِن تَتَّبِعُونَ إِلَّا رَجُلًۭا مَّسْحُورًا ‎</p>
 </td>
 <td width="50%">'ʔaw yulqay'ʔ 'ʔilayhi kanzun 'ʔaw takuwnu lahuw jan:atunm ya'ʔkulu minha' j waqa'la 'lẓ:alimuwna 'ʔin tat:abiʕuwna 'ʔil:a' rajulanm' m:asḥuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_009">9</a> ‏ٱنظُرْ كَيْفَ ضَرَبُوا۟ لَكَ ٱلْأَمْثَلَ فَضَلُّوا۟ فَلَا يَسْتَطِيعُونَ سَبِيلًۭا ‎</p>
 </td>
 <td width="50%">'nẓur kayfa ḍarabuw' laka 'l'ʔamθala faḍal:uw' fala' yastaṭiyʕuwna sabiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_010">10</a> ‏تَبَارَكَ ٱلَّذِىٓ إِن شَآءَ جَعَلَ لَكَ خَيْرًۭا مِّن ذَلِكَ جَنَّتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَرُ وَيَجْعَل لَّكَ قُصُورًۢا ‎</p>
 </td>
 <td width="50%">taba'raka 'l:aðiyʔ 'ʔin ʃa'ʔʔa jaʕala laka xayranm' m:in ðalika jan:atinm tajriy min taḥtiha' 'l'ʔanharu wayajʕal l:aka quṣuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_011">11</a> ‏بَلْ كَذَّبُوا۟ بِٱلسَّاعَةِ ۖ وَأَعْتَدْنَا لِمَن كَذَّبَ بِٱلسَّاعَةِ سَعِيرًا ‎</p>
 </td>
 <td width="50%">bal kað:abuw' bi'ls:a'ʕati sl' wa'ʔaʕtadna' liman kað:aba bi'ls:a'ʕati saʕiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_012">12</a> ‏إِذَا رَأَتْهُم مِّن مَّكَانٍۭ بَعِيدٍۢ سَمِعُوا۟ لَهَا تَغَيُّظًۭا وَزَفِيرًۭا ‎</p>
 </td>
 <td width="50%">'ʔiða' ra'ʔathum m:in m:aka'ninm baʕiydinm samiʕuw' laha' taɣay:uẓanm' wazafiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_013">13</a> ‏وَإِذَآ أُلْقُوا۟ مِنْهَا مَكَانًۭا ضَيِّقًۭا مُّقَرَّنِينَ دَعَوْا۟ هُنَالِكَ ثُبُورًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔiða'ʔ 'ʔulquw' minha' maka'nanm' ḍay:iqanm' m:uqar:aniyna daʕaw' huna'lika θubuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_014">14</a> ‏لَّا تَدْعُوا۟ ٱلْيَوْمَ ثُبُورًۭا وَحِدًۭا وَٱدْعُوا۟ ثُبُورًۭا كَثِيرًۭا ‎</p>
 </td>
 <td width="50%">l:a' tadʕuw' 'lyawma θubuwranm' waḥidanm' wa'dʕuw' θubuwranm' kaθiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_015">15</a> ‏قُلْ أَذَلِكَ خَيْرٌ أَمْ جَنَّةُ ٱلْخُلْدِ ٱلَّتِى وُعِدَ ٱلْمُتَّقُونَ ۚ كَانَتْ لَهُمْ جَزَآءًۭ وَمَصِيرًۭا ‎</p>
 </td>
 <td width="50%">qul 'ʔaðalika xayrun 'ʔam jan:atu 'lxuldi 'l:atiy wuʕida 'lmut:aquwna j ka'nat lahum jaza'ʔʔanm wamaṣiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_016">16</a> ‏لَّهُمْ فِيهَا مَا يَشَآءُونَ خَلِدِينَ ۚ كَانَ عَلَىٰ رَبِّكَ وَعْدًۭا مَّسْـُٔولًۭا ‎</p>
 </td>
 <td width="50%">l:ahum fiyha' ma' yaʃa'ʔʔuwna xalidiyna j ka'na ʕalay' rab:ika waʕdanm' m:asʔuwlanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_017">17</a> ‏وَيَوْمَ يَحْشُرُهُمْ وَمَا يَعْبُدُونَ مِن دُونِ ٱللَّهِ فَيَقُولُ ءَأَنتُمْ أَضْلَلْتُمْ عِبَادِى هَٓؤُلَآءِ أَمْ هُمْ ضَلُّوا۟ ٱلسَّبِيلَ ‎</p>
 </td>
 <td width="50%">wayawma yaḥʃuruhum wama' yaʕbuduwna min duwni 'll:ahi fayaquwlu ʔa'ʔantum 'ʔaḍlaltum ʕiba'diy haʔwʔula'ʔʔi 'ʔam hum ḍal:uw' 'ls:abiyla </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_018">18</a> ‏قَالُوا۟ سُبْحَنَكَ مَا كَانَ يَنۢبَغِى لَنَآ أَن نَّتَّخِذَ مِن دُونِكَ مِنْ أَوْلِيَآءَ وَلَكِن مَّتَّعْتَهُمْ وَءَابَآءَهُمْ حَتَّىٰ نَسُوا۟ ٱلذِّكْرَ وَكَانُوا۟ قَوْمًۢا بُورًۭا ‎</p>
 </td>
 <td width="50%">qa'luw' subḥanaka ma' ka'na yanmbaɣiy lana'ʔ 'ʔan n:at:axiða min duwnika min 'ʔawliya'ʔʔa walakin m:at:aʕtahum waʔa'ba'ʔʔahum ḥat:ay' nasuw' 'lð:ikra waka'nuw' qawmanm' buwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_019">19</a> ‏فَقَدْ كَذَّبُوكُم بِمَا تَقُولُونَ فَمَا تَسْتَطِيعُونَ صَرْفًۭا وَلَا نَصْرًۭا ۚ وَمَن يَظْلِم مِّنكُمْ نُذِقْهُ عَذَابًۭا كَبِيرًۭا ‎</p>

 <td width="50%">faqad kað:abuwkum bima' taquwluwna fama' tastaṭiyʕuwna ṣarfanm' wala' naṣranm' j waman yaẓlim m:inkum nuðiqhu ʕaða'banm' kabiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_020">20</a> ‏وَمَآ أَرْسَلْنَا قَبْلَكَ مِنَ ٱلْمُرْسَلِينَ إِلَّآ إِنَّهُمْ لَيَأْكُلُونَ ٱلطَّعَامَ وَيَمْشُونَ فِى ٱلْأَسْوَاقِ ۗ وَجَعَلْنَا بَعْضَكُمْ لِبَعْضٍۢ فِتْنَةً أَتَصْبِرُونَ ۗ وَكَانَ رَبُّكَ بَصِيرًۭا ‎</p>
 </td>
 <td width="50%">wama'ʔ 'ʔarsalna' qablaka mina 'lmursaliyna 'ʔil:a'ʔ 'ʔin:ahum laya'ʔkuluwna 'lṭ:aʕa'ma wayamʃuwna fiy 'l'ʔaswa'qi ql' wajaʕalna' baʕḍakum libaʕḍinm fitnatan 'ʔataṣbiruwna ql' waka'na rab:uka baṣiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_021">21</a> ‏ وَقَالَ ٱلَّذِينَ لَا يَرْجُونَ لِقَآءَنَا لَوْلَآ أُنزِلَ عَلَيْنَا ٱلْمَلَٓئِكَةُ أَوْ نَرَىٰ رَبَّنَا ۗ لَقَدِ ٱسْتَكْبَرُوا۟ فِىٓ أَنفُسِهِمْ وَعَتَوْ عُتُوًّۭا كَبِيرًۭا ‎</p>
 </td>
 <td width="50%"> waqa'la 'l:aðiyna la' yarjuwna liqa'ʔʔana' lawla'ʔ 'ʔunzila ʕalayna' 'lmalaʔyʔikatu 'ʔaw naray' rab:ana' ql' laqadi 'stakbaruw' fiyʔ 'ʔanfusihim waʕataw ʕutuw:anm' kabiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_022">22</a> ‏يَوْمَ يَرَوْنَ ٱلْمَلَٓئِكَةَ لَا بُشْرَىٰ يَوْمَئِذٍۢ لِّلْمُجْرِمِينَ وَيَقُولُونَ حِجْرًۭا مَّحْجُورًۭا ‎</p>
 </td>
 <td width="50%">yawma yarawna 'lmalaʔyʔikata la' buʃray' yawmayʔiðinm l:ilmujrimiyna wayaquwluwna ḥijranm' m:aḥjuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_023">23</a> ‏وَقَدِمْنَآ إِلَىٰ مَا عَمِلُوا۟ مِنْ عَمَلٍۢ فَجَعَلْنَهُ هَبَآءًۭ مَّنثُورًا ‎</p>
 </td>
 <td width="50%">waqadimna'ʔ 'ʔilay' ma' ʕamiluw' min ʕamalinm fajaʕalnahu haba'ʔʔanm m:anθuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_024">24</a> ‏أَصْحَبُ ٱلْجَنَّةِ يَوْمَئِذٍ خَيْرٌۭ مُّسْتَقَرًّۭا وَأَحْسَنُ مَقِيلًۭا ‎</p>
 </td>

 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_025">25</a> ‏وَيَوْمَ تَشَقَّقُ ٱلسَّمَآءُ بِٱلْغَمَمِ وَنُزِّلَ ٱلْمَلَٓئِكَةُ تَنزِيلًا ‎</p>
 </td>
 <td width="50%">wayawma taʃaq:aqu 'ls:ama'ʔʔu bi'lɣamami wanuz:ila 'lmalaʔyʔikatu tanziylan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_026">26</a> ‏ٱلْمُلْكُ يَوْمَئِذٍ ٱلْحَقُّ لِلرَّحْمَنِ ۚ وَكَانَ يَوْمًا عَلَى ٱلْكَفِرِينَ عَسِيرًۭا ‎</p>
 </td>
 <td width="50%">'lmulku yawmayʔiðin 'lḥaq:u lilr:aḥmani j waka'na yawman' ʕalay 'lkafiriyna ʕasiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_027">27</a> ‏وَيَوْمَ يَعَضُّ ٱلظَّالِمُ عَلَىٰ يَدَيْهِ يَقُولُ يَلَيْتَنِى ٱتَّخَذْتُ مَعَ ٱلرَّسُولِ سَبِيلًۭا ‎</p>
 </td>
 <td width="50%">wayawma yaʕaḍ:u 'lẓ:a'limu ʕalay' yadayhi yaquwlu yalaytaniy 't:axaðtu maʕa 'lr:asuwli sabiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_025_028">28</a> ‏يَوَيْلَتَىٰ لَيْتَنِى لَمْ أَتَّخِذْ فُلَانًا خَلِيلًۭا ‎</p>
 </td>

 </tr>
 <tr valign="TOP">

 </td>
 <td width="50%">l:aqad 'ʔaḍal:aniy ʕani 'lð:ikri baʕda 'ʔið ja'ʔʔaniy ql' waka'na 'lʃ:ayṭanu lil'ʔinsani xaðuwlanm' </td>
