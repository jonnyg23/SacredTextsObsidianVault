 <h1 align="CENTER">71. ‏نوح‎</h1>
 <h2 align="CENTER">nwḥ</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_001">1</a> ‏إِنَّآ أَرْسَلْنَا نُوحًا إِلَىٰ قَوْمِهِۦٓ أَنْ أَنذِرْ قَوْمَكَ مِن قَبْلِ أَن يَأْتِيَهُمْ عَذَابٌ أَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">'ʔin:a'ʔ 'ʔarsalna' nuwḥan' 'ʔilay' qawmihiyʔ 'ʔan 'ʔanðir qawmaka min qabli 'ʔan ya'ʔtiyahum ʕaða'bun 'ʔaliymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_002">2</a> ‏قَالَ يَقَوْمِ إِنِّى لَكُمْ نَذِيرٌۭ مُّبِينٌ ‎</p>
 </td>
 <td width="50%">qa'la yaqawmi 'ʔin:iy lakum naðiyrunm m:ubiynun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_003">3</a> ‏أَنِ ٱعْبُدُوا۟ ٱللَّهَ وَٱتَّقُوهُ وَأَطِيعُونِ ‎</p>
 </td>
 <td width="50%">'ʔani 'ʕbuduw' 'll:aha wa't:aquwhu wa'ʔaṭiyʕuwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_004">4</a> ‏يَغْفِرْ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرْكُمْ إِلَىٰٓ أَجَلٍۢ مُّسَمًّى ۚ إِنَّ أَجَلَ ٱللَّهِ إِذَا جَآءَ لَا يُؤَخَّرُ ۖ لَوْ كُنتُمْ تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">yaɣfir lakum m:in ðunuwbikum wayuwʔax:irkum 'ʔilay'ʔ 'ʔajalinm m:usam:any j 'ʔin:a 'ʔajala 'll:ahi 'ʔiða' ja'ʔʔa la' yuwʔax:aru sl' law kuntum taʕlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_005">5</a> ‏قَالَ رَبِّ إِنِّى دَعَوْتُ قَوْمِى لَيْلًۭا وَنَهَارًۭا ‎</p>
 </td>
 <td width="50%">qa'la rab:i 'ʔin:iy daʕawtu qawmiy laylanm' wanaha'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_006">6</a> ‏فَلَمْ يَزِدْهُمْ دُعَآءِىٓ إِلَّا فِرَارًۭا ‎</p>
 </td>
 <td width="50%">falam yazidhum duʕa'ʔʔiyʔ 'ʔil:a' fira'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_007">7</a> ‏وَإِنِّى كُلَّمَا دَعَوْتُهُمْ لِتَغْفِرَ لَهُمْ جَعَلُوٓا۟ أَصَبِعَهُمْ فِىٓ ءَاذَانِهِمْ وَٱسْتَغْشَوْا۟ ثِيَابَهُمْ وَأَصَرُّوا۟ وَٱسْتَكْبَرُوا۟ ٱسْتِكْبَارًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔin:iy kul:ama' daʕawtuhum litaɣfira lahum jaʕaluwʔ' 'ʔaṣabiʕahum fiyʔ ʔa'ða'nihim wa'staɣʃaw' θiya'bahum wa'ʔaṣar:uw' wa'stakbaruw' 'stikba'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_008">8</a> ‏ثُمَّ إِنِّى دَعَوْتُهُمْ جِهَارًۭا ‎</p>
 </td>
 <td width="50%">θum:a 'ʔin:iy daʕawtuhum jiha'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_009">9</a> ‏ثُمَّ إِنِّىٓ أَعْلَنتُ لَهُمْ وَأَسْرَرْتُ لَهُمْ إِسْرَارًۭا ‎</p>
 </td>
 <td width="50%">θum:a 'ʔin:iyʔ 'ʔaʕlantu lahum wa'ʔasrartu lahum 'ʔisra'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_010">10</a> ‏فَقُلْتُ ٱسْتَغْفِرُوا۟ رَبَّكُمْ إِنَّهُۥ كَانَ غَفَّارًۭا ‎</p>
 </td>
 <td width="50%">faqultu 'staɣfiruw' rab:akum 'ʔin:ahuw ka'na ɣaf:a'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_011">11</a> ‏يُرْسِلِ ٱلسَّمَآءَ عَلَيْكُم مِّدْرَارًۭا ‎</p>
 </td>
 <td width="50%">yursili 'ls:ama'ʔʔa ʕalaykum m:idra'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_012">12</a> ‏وَيُمْدِدْكُم بِأَمْوَلٍۢ وَبَنِينَ وَيَجْعَل لَّكُمْ جَنَّتٍۢ وَيَجْعَل لَّكُمْ أَنْهَرًۭا ‎</p>
 </td>
 <td width="50%">wayumdidkum bi'ʔamwalinm wabaniyna wayajʕal l:akum jan:atinm wayajʕal l:akum 'ʔanharanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_013">13</a> ‏مَّا لَكُمْ لَا تَرْجُونَ لِلَّهِ وَقَارًۭا ‎</p>
 </td>
 <td width="50%">m:a' lakum la' tarjuwna lil:ahi waqa'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_014">14</a> ‏وَقَدْ خَلَقَكُمْ أَطْوَارًا ‎</p>
 </td>
 <td width="50%">waqad xalaqakum 'ʔaṭwa'ran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_015">15</a> ‏أَلَمْ تَرَوْا۟ كَيْفَ خَلَقَ ٱللَّهُ سَبْعَ سَمَوَتٍۢ طِبَاقًۭا ‎</p>
 </td>
 <td width="50%">'ʔalam taraw' kayfa xalaqa 'll:ahu sabʕa samawatinm ṭiba'qanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_016">16</a> ‏وَجَعَلَ ٱلْقَمَرَ فِيهِنَّ نُورًۭا وَجَعَلَ ٱلشَّمْسَ سِرَاجًۭا ‎</p>
 </td>
 <td width="50%">wajaʕala 'lqamara fiyhin:a nuwranm' wajaʕala 'lʃ:amsa sira'janm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_017">17</a> ‏وَٱللَّهُ أَنۢبَتَكُم مِّنَ ٱلْأَرْضِ نَبَاتًۭا ‎</p>
 </td>
 <td width="50%">wa'll:ahu 'ʔanmbatakum m:ina 'l'ʔarḍi naba'tanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_018">18</a> ‏ثُمَّ يُعِيدُكُمْ فِيهَا وَيُخْرِجُكُمْ إِخْرَاجًۭا ‎</p>
 </td>
 <td width="50%">θum:a yuʕiydukum fiyha' wayuxrijukum 'ʔixra'janm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_019">19</a> ‏وَٱللَّهُ جَعَلَ لَكُمُ ٱلْأَرْضَ بِسَاطًۭا ‎</p>
 </td>
 <td width="50%">wa'll:ahu jaʕala lakumu 'l'ʔarḍa bisa'ṭanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_020">20</a> ‏لِّتَسْلُكُوا۟ مِنْهَا سُبُلًۭا فِجَاجًۭا ‎</p>
 </td>
 <td width="50%">l:itaslukuw' minha' subulanm' fija'janm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_021">21</a> ‏قَالَ نُوحٌۭ رَّبِّ إِنَّهُمْ عَصَوْنِى وَٱتَّبَعُوا۟ مَن لَّمْ يَزِدْهُ مَالُهُۥ وَوَلَدُهُۥٓ إِلَّا خَسَارًۭا ‎</p>
 </td>
 <td width="50%">qa'la nuwḥunm r:ab:i 'ʔin:ahum ʕaṣawniy wa't:abaʕuw' man l:am yazidhu ma'luhuw wawaladuhuwʔ 'ʔil:a' xasa'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_022">22</a> ‏وَمَكَرُوا۟ مَكْرًۭا كُبَّارًۭا ‎</p>
 </td>
 <td width="50%">wamakaruw' makranm' kub:a'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_023">23</a> ‏وَقَالُوا۟ لَا تَذَرُنَّ ءَالِهَتَكُمْ وَلَا تَذَرُنَّ وَدًّۭا وَلَا سُوَاعًۭا وَلَا يَغُوثَ وَيَعُوقَ وَنَسْرًۭا ‎</p>
 </td>
 <td width="50%">waqa'luw' la' taðarun:a ʔa'lihatakum wala' taðarun:a wad:anm' wala' suwa'ʕanm' wala' yaɣuwθa wayaʕuwqa wanasranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_024">24</a> ‏وَقَدْ أَضَلُّوا۟ كَثِيرًۭا ۖ وَلَا تَزِدِ ٱلظَّلِمِينَ إِلَّا ضَلَلًۭا ‎</p>
 </td>
 <td width="50%">waqad 'ʔaḍal:uw' kaθiyranm' sl' wala' tazidi 'lẓ:alimiyna 'ʔil:a' ḍalalanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_025">25</a> ‏مِّمَّا خَطِيٓـَٔتِهِمْ أُغْرِقُوا۟ فَأُدْخِلُوا۟ نَارًۭا فَلَمْ يَجِدُوا۟ لَهُم مِّن دُونِ ٱللَّهِ أَنصَارًۭا ‎</p>
 </td>
 <td width="50%">m:im:a' xaṭiyʔʔatihim 'ʔuɣriquw' fa'ʔudxiluw' na'ranm' falam yajiduw' lahum m:in duwni 'll:ahi 'ʔanṣa'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_026">26</a> ‏وَقَالَ نُوحٌۭ رَّبِّ لَا تَذَرْ عَلَى ٱلْأَرْضِ مِنَ ٱلْكَفِرِينَ دَيَّارًا ‎</p>
 </td>
 <td width="50%">waqa'la nuwḥunm r:ab:i la' taðar ʕalay 'l'ʔarḍi mina 'lkafiriyna day:a'ran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_027">27</a> ‏إِنَّكَ إِن تَذَرْهُمْ يُضِلُّوا۟ عِبَادَكَ وَلَا يَلِدُوٓا۟ إِلَّا فَاجِرًۭا كَفَّارًۭا ‎</p>
 </td>
 <td width="50%">'ʔin:aka 'ʔin taðarhum yuḍil:uw' ʕiba'daka wala' yaliduwʔ' 'ʔil:a' fa'jiranm' kaf:a'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_071_028">28</a> ‏رَّبِّ ٱغْفِرْ لِى وَلِوَلِدَىَّ وَلِمَن دَخَلَ بَيْتِىَ مُؤْمِنًۭا وَلِلْمُؤْمِنِينَ وَٱلْمُؤْمِنَتِ وَلَا تَزِدِ ٱلظَّلِمِينَ إِلَّا تَبَارًۢا ‎</p>
 </td>
 <td width="50%">r:ab:i 'ɣfir liy waliwaliday:a waliman daxala baytiya muwʔminanm' walilmuwʔminiyna wa'lmuwʔminati wala' tazidi 'lẓ:alimiyna 'ʔil:a' taba'ranm' </td>
 </tr>
 </table>
