 <h1 align="CENTER">62. ‏الجمعة‎</h1>
 <h2 align="CENTER">'ljmʕt</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_001">1</a> ‏يُسَبِّحُ لِلَّهِ مَا فِى ٱلسَّمَوَتِ وَمَا فِى ٱلْأَرْضِ ٱلْمَلِكِ ٱلْقُدُّوسِ ٱلْعَزِيزِ ٱلْحَكِيمِ ‎</p>
 </td>
 <td width="50%">yusab:iḥu lil:ahi ma' fiy 'ls:amawati wama' fiy 'l'ʔarḍi 'lmaliki 'lqud:uwsi 'lʕaziyzi 'lḥakiymi </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_002">2</a> ‏هُوَ ٱلَّذِى بَعَثَ فِى ٱلْأُمِّيِّۦنَ رَسُولًۭا مِّنْهُمْ يَتْلُوا۟ عَلَيْهِمْ ءَايَتِهِۦ وَيُزَكِّيهِمْ وَيُعَلِّمُهُمُ ٱلْكِتَبَ وَٱلْحِكْمَةَ وَإِن كَانُوا۟ مِن قَبْلُ لَفِى ضَلَلٍۢ مُّبِينٍۢ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiy baʕaθa fiy 'l'ʔum:iy:iyna rasuwlanm' m:inhum yatluw' ʕalayhim ʔa'yatihiy wayuzak:iyhim wayuʕal:imuhumu 'lkitaba wa'lḥikmata wa'ʔin ka'nuw' min qablu lafiy ḍalalinm m:ubiyninm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_003">3</a> ‏وَءَاخَرِينَ مِنْهُمْ لَمَّا يَلْحَقُوا۟ بِهِمْ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">waʔa'xariyna minhum lam:a' yalḥaquw' bihim j wahuwa 'lʕaziyzu 'lḥakiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_004">4</a> ‏ذَلِكَ فَضْلُ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُ ۚ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ ‎</p>
 </td>
 <td width="50%">ðalika faḍlu 'll:ahi yuwʔtiyhi man yaʃa'ʔʔu j wa'll:ahu ðuw 'lfaḍli 'lʕaẓiymi </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_005">5</a> ‏مَثَلُ ٱلَّذِينَ حُمِّلُوا۟ ٱلتَّوْرَىٰةَ ثُمَّ لَمْ يَحْمِلُوهَا كَمَثَلِ ٱلْحِمَارِ يَحْمِلُ أَسْفَارًۢا ۚ بِئْسَ مَثَلُ ٱلْقَوْمِ ٱلَّذِينَ كَذَّبُوا۟ بِـَٔايَتِ ٱللَّهِ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%">maθalu 'l:aðiyna ḥum:iluw' 'lt:awray'ta θum:a lam yaḥmiluwha' kamaθali 'lḥima'ri yaḥmilu 'ʔasfa'ranm' j biyʔsa maθalu 'lqawmi 'l:aðiyna kað:abuw' biʔa'yati 'll:ahi j wa'll:ahu la' yahdiy 'lqawma 'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_006">6</a> ‏قُلْ يَٓأَيُّهَا ٱلَّذِينَ هَادُوٓا۟ إِن زَعَمْتُمْ أَنَّكُمْ أَوْلِيَآءُ لِلَّهِ مِن دُونِ ٱلنَّاسِ فَتَمَنَّوُا۟ ٱلْمَوْتَ إِن كُنتُمْ صَدِقِينَ ‎</p>
 </td>
 <td width="50%">qul yaʔ'ʔay:uha' 'l:aðiyna ha'duwʔ' 'ʔin zaʕamtum 'ʔan:akum 'ʔawliya'ʔʔu lil:ahi min duwni 'ln:a'si fataman:awu' 'lmawta 'ʔin kuntum ṣadiqiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_007">7</a> ‏وَلَا يَتَمَنَّوْنَهُۥٓ أَبَدًۢا بِمَا قَدَّمَتْ أَيْدِيهِمْ ۚ وَٱللَّهُ عَلِيمٌۢ بِٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%">wala' yataman:awnahuwʔ 'ʔabadanm' bima' qad:amat 'ʔaydiyhim j wa'll:ahu ʕaliymunm bi'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_008">8</a> ‏قُلْ إِنَّ ٱلْمَوْتَ ٱلَّذِى تَفِرُّونَ مِنْهُ فَإِنَّهُۥ مُلَقِيكُمْ ۖ ثُمَّ تُرَدُّونَ إِلَىٰ عَلِمِ ٱلْغَيْبِ وَٱلشَّهَدَةِ فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ‎</p>
 </td>
 <td width="50%">qul 'ʔin:a 'lmawta 'l:aðiy tafir:uwna minhu fa'ʔin:ahuw mulaqiykum sl' θum:a turad:uwna 'ʔilay' ʕalimi 'lɣaybi wa'lʃ:ahadati fayunab:iyʔukum bima' kuntum taʕmaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_009">9</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا نُودِىَ لِلصَّلَوٰةِ مِن يَوْمِ ٱلْجُمُعَةِ فَٱسْعَوْا۟ إِلَىٰ ذِكْرِ ٱللَّهِ وَذَرُوا۟ ٱلْبَيْعَ ۚ ذَلِكُمْ خَيْرٌۭ لَّكُمْ إِن كُنتُمْ تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuwʔ' 'ʔiða' nuwdiya lilṣ:alaw'ti min yawmi 'ljumuʕati fa'sʕaw' 'ʔilay' ðikri 'll:ahi waðaruw' 'lbayʕa j ðalikum xayrunm l:akum 'ʔin kuntum taʕlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_010">10</a> ‏فَإِذَا قُضِيَتِ ٱلصَّلَوٰةُ فَٱنتَشِرُوا۟ فِى ٱلْأَرْضِ وَٱبْتَغُوا۟ مِن فَضْلِ ٱللَّهِ وَٱذْكُرُوا۟ ٱللَّهَ كَثِيرًۭا لَّعَلَّكُمْ تُفْلِحُونَ ‎</p>
 </td>
 <td width="50%">fa'ʔiða' quḍiyati 'lṣ:alaw'tu fa'ntaʃiruw' fiy 'l'ʔarḍi wa'btaɣuw' min faḍli 'll:ahi wa'ðkuruw' 'll:aha kaθiyranm' l:aʕal:akum tufliḥuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_062_011">11</a> ‏وَإِذَا رَأَوْا۟ تِجَرَةً أَوْ لَهْوًا ٱنفَضُّوٓا۟ إِلَيْهَا وَتَرَكُوكَ قَآئِمًۭا ۚ قُلْ مَا عِندَ ٱللَّهِ خَيْرٌۭ مِّنَ ٱللَّهْوِ وَمِنَ ٱلتِّجَرَةِ ۚ وَٱللَّهُ خَيْرُ ٱلرَّزِقِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔiða' ra'ʔaw' tijaratan 'ʔaw lahwan' 'nfaḍ:uwʔ' 'ʔilayha' watarakuwka qa'ʔyʔimanm' j qul ma' ʕinda 'll:ahi xayrunm m:ina 'll:ahwi wamina 'lt:ijarati j wa'll:ahu xayru 'lr:aziqiyna </td>
 </tr>
