 <h1 align="CENTER">6. ‏الأنعام‎</h1>
 <h2 align="CENTER">'l'ʔnʕ'm</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_001">1</a> ‏ٱلْحَمْدُ لِلَّهِ ٱلَّذِى خَلَقَ ٱلسَّمَوَتِ وَٱلْأَرْضَ وَجَعَلَ ٱلظُّلُمَتِ وَٱلنُّورَ ۖ ثُمَّ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ يَعْدِلُونَ ‎</p>
 </td>
 <td width="50%">'lḥamdu lil:ahi 'l:aðiy xalaqa 'ls:amawati wa'l'ʔarḍa wajaʕala 'lẓ:ulumati wa'ln:uwra sl' θum:a 'l:aðiyna kafaruw' birab:ihim yaʕdiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_002">2</a> ‏هُوَ ٱلَّذِى خَلَقَكُم مِّن طِينٍۢ ثُمَّ قَضَىٰٓ أَجَلًۭا ۖ وَأَجَلٌۭ مُّسَمًّى عِندَهُۥ ۖ ثُمَّ أَنتُمْ تَمْتَرُونَ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiy xalaqakum m:in ṭiyninm θum:a qaḍay'ʔ 'ʔajalanm' sl' wa'ʔajalunm m:usam:any ʕindahuw sl' θum:a 'ʔantum tamtaruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_003">3</a> ‏وَهُوَ ٱللَّهُ فِى ٱلسَّمَوَتِ وَفِى ٱلْأَرْضِ ۖ يَعْلَمُ سِرَّكُمْ وَجَهْرَكُمْ وَيَعْلَمُ مَا تَكْسِبُونَ ‎</p>
 </td>
 <td width="50%">wahuwa 'll:ahu fiy 'ls:amawati wafiy 'l'ʔarḍi sl' yaʕlamu sir:akum wajahrakum wayaʕlamu ma' taksibuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_004">4</a> ‏وَمَا تَأْتِيهِم مِّنْ ءَايَةٍۢ مِّنْ ءَايَتِ رَبِّهِمْ إِلَّا كَانُوا۟ عَنْهَا مُعْرِضِينَ ‎</p>
 </td>
 <td width="50%">wama' ta'ʔtiyhim m:in ʔa'yatinm m:in ʔa'yati rab:ihim 'ʔil:a' ka'nuw' ʕanha' muʕriḍiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_005">5</a> ‏فَقَدْ كَذَّبُوا۟ بِٱلْحَقِّ لَمَّا جَآءَهُمْ ۖ فَسَوْفَ يَأْتِيهِمْ أَنۢبَٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ‎</p>
 </td>
 <td width="50%">faqad kað:abuw' bi'lḥaq:i lam:a' ja'ʔʔahum sl' fasawfa ya'ʔtiyhim 'ʔanmbaʔwʔu' ma' ka'nuw' bihiy yastahziʔuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_006">6</a> ‏أَلَمْ يَرَوْا۟ كَمْ أَهْلَكْنَا مِن قَبْلِهِم مِّن قَرْنٍۢ مَّكَّنَّهُمْ فِى ٱلْأَرْضِ مَا لَمْ نُمَكِّن لَّكُمْ وَأَرْسَلْنَا ٱلسَّمَآءَ عَلَيْهِم مِّدْرَارًۭا وَجَعَلْنَا ٱلْأَنْهَرَ تَجْرِى مِن تَحْتِهِمْ فَأَهْلَكْنَهُم بِذُنُوبِهِمْ وَأَنشَأْنَا مِنۢ بَعْدِهِمْ قَرْنًا ءَاخَرِينَ ‎</p>
 </td>
 <td width="50%">'ʔalam yaraw' kam 'ʔahlakna' min qablihim m:in qarninm m:ak:an:ahum fiy 'l'ʔarḍi ma' lam numak:in l:akum wa'ʔarsalna' 'ls:ama'ʔʔa ʕalayhim m:idra'ranm' wajaʕalna' 'l'ʔanhara tajriy min taḥtihim fa'ʔahlaknahum biðunuwbihim wa'ʔanʃa'ʔna' minm baʕdihim qarnan' ʔa'xariyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_007">7</a> ‏وَلَوْ نَزَّلْنَا عَلَيْكَ كِتَبًۭا فِى قِرْطَاسٍۢ فَلَمَسُوهُ بِأَيْدِيهِمْ لَقَالَ ٱلَّذِينَ كَفَرُوٓا۟ إِنْ هَذَآ إِلَّا سِحْرٌۭ مُّبِينٌۭ ‎</p>
 </td>
 <td width="50%">walaw naz:alna' ʕalayka kitabanm' fiy qirṭa'sinm falamasuwhu bi'ʔaydiyhim laqa'la 'l:aðiyna kafaruwʔ' 'ʔin haða'ʔ 'ʔil:a' siḥrunm m:ubiynunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_008">8</a> ‏وَقَالُوا۟ لَوْلَآ أُنزِلَ عَلَيْهِ مَلَكٌۭ ۖ وَلَوْ أَنزَلْنَا مَلَكًۭا لَّقُضِىَ ٱلْأَمْرُ ثُمَّ لَا يُنظَرُونَ ‎</p>
 </td>
 <td width="50%">waqa'luw' lawla'ʔ 'ʔunzila ʕalayhi malakunm sl' walaw 'ʔanzalna' malakanm' l:aquḍiya 'l'ʔamru θum:a la' yunẓaruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_009">9</a> ‏وَلَوْ جَعَلْنَهُ مَلَكًۭا لَّجَعَلْنَهُ رَجُلًۭا وَلَلَبَسْنَا عَلَيْهِم مَّا يَلْبِسُونَ ‎</p>
 </td>
 <td width="50%">walaw jaʕalnahu malakanm' l:ajaʕalnahu rajulanm' walalabasna' ʕalayhim m:a' yalbisuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_010">10</a> ‏وَلَقَدِ ٱسْتُهْزِئَ بِرُسُلٍۢ مِّن قَبْلِكَ فَحَاقَ بِٱلَّذِينَ سَخِرُوا۟ مِنْهُم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ‎</p>
 </td>
 <td width="50%">walaqadi 'stuhziyʔa birusulinm m:in qablika faḥa'qa bi'l:aðiyna saxiruw' minhum m:a' ka'nuw' bihiy yastahziʔuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_011">11</a> ‏قُلْ سِيرُوا۟ فِى ٱلْأَرْضِ ثُمَّ ٱنظُرُوا۟ كَيْفَ كَانَ عَقِبَةُ ٱلْمُكَذِّبِينَ ‎</p>
 </td>
 <td width="50%">qul siyruw' fiy 'l'ʔarḍi θum:a 'nẓuruw' kayfa ka'na ʕaqibatu 'lmukað:ibiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_012">12</a> ‏قُل لِّمَن مَّا فِى ٱلسَّمَوَتِ وَٱلْأَرْضِ ۖ قُل لِّلَّهِ ۚ كَتَبَ عَلَىٰ نَفْسِهِ ٱلرَّحْمَةَ ۚ لَيَجْمَعَنَّكُمْ إِلَىٰ يَوْمِ ٱلْقِيَمَةِ لَا رَيْبَ فِيهِ ۚ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُمْ فَهُمْ لَا يُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">qul l:iman m:a' fiy 'ls:amawati wa'l'ʔarḍi sl' qul l:il:ahi j kataba ʕalay' nafsihi 'lr:aḥmata j layajmaʕan:akum 'ʔilay' yawmi 'lqiyamati la' rayba fiyhi j 'l:aðiyna xasiruwʔ' 'ʔanfusahum fahum la' yuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_013">13</a> ‏ وَلَهُۥ مَا سَكَنَ فِى ٱلَّيْلِ وَٱلنَّهَارِ ۚ وَهُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ‎</p>
 </td>
 <td width="50%"> walahuw ma' sakana fiy 'l:ayli wa'ln:aha'ri j wahuwa 'ls:amiyʕu 'lʕaliymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_014">14</a> ‏قُلْ أَغَيْرَ ٱللَّهِ أَتَّخِذُ وَلِيًّۭا فَاطِرِ ٱلسَّمَوَتِ وَٱلْأَرْضِ وَهُوَ يُطْعِمُ وَلَا يُطْعَمُ ۗ قُلْ إِنِّىٓ أُمِرْتُ أَنْ أَكُونَ أَوَّلَ مَنْ أَسْلَمَ ۖ وَلَا تَكُونَنَّ مِنَ ٱلْمُشْرِكِينَ ‎</p>
 </td>
 <td width="50%">qul 'ʔaɣayra 'll:ahi 'ʔat:axiðu waliy:anm' fa'ṭiri 'ls:amawati wa'l'ʔarḍi wahuwa yuṭʕimu wala' yuṭʕamu ql' qul 'ʔin:iyʔ 'ʔumirtu 'ʔan 'ʔakuwna 'ʔaw:ala man 'ʔaslama sl' wala' takuwnan:a mina 'lmuʃrikiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_015">15</a> ‏قُلْ إِنِّىٓ أَخَافُ إِنْ عَصَيْتُ رَبِّى عَذَابَ يَوْمٍ عَظِيمٍۢ ‎</p>
 </td>
 <td width="50%">qul 'ʔin:iyʔ 'ʔaxa'fu 'ʔin ʕaṣaytu rab:iy ʕaða'ba yawmin ʕaẓiyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_016">16</a> ‏مَّن يُصْرَفْ عَنْهُ يَوْمَئِذٍۢ فَقَدْ رَحِمَهُۥ ۚ وَذَلِكَ ٱلْفَوْزُ ٱلْمُبِينُ ‎</p>
 </td>
 <td width="50%">m:an yuṣraf ʕanhu yawmayʔiðinm faqad raḥimahuw j waðalika 'lfawzu 'lmubiynu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_017">17</a> ‏وَإِن يَمْسَسْكَ ٱللَّهُ بِضُرٍّۢ فَلَا كَاشِفَ لَهُۥٓ إِلَّا هُوَ ۖ وَإِن يَمْسَسْكَ بِخَيْرٍۢ فَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌۭ ‎</p>
 </td>
 <td width="50%">wa'ʔin yamsaska 'll:ahu biḍur:inm fala' ka'ʃifa lahuwʔ 'ʔil:a' huwa sl' wa'ʔin yamsaska bixayrinm fahuwa ʕalay' kul:i ʃayʔinm qadiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_018">18</a> ‏وَهُوَ ٱلْقَاهِرُ فَوْقَ عِبَادِهِۦ ۚ وَهُوَ ٱلْحَكِيمُ ٱلْخَبِيرُ ‎</p>
 </td>
 <td width="50%">wahuwa 'lqa'hiru fawqa ʕiba'dihiy j wahuwa 'lḥakiymu 'lxabiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_019">19</a> ‏قُلْ أَىُّ شَىْءٍ أَكْبَرُ شَهَدَةًۭ ۖ قُلِ ٱللَّهُ ۖ شَهِيدٌۢ بَيْنِى وَبَيْنَكُمْ ۚ وَأُوحِىَ إِلَىَّ هَذَا ٱلْقُرْءَانُ لِأُنذِرَكُم بِهِۦ وَمَنۢ بَلَغَ ۚ أَئِنَّكُمْ لَتَشْهَدُونَ أَنَّ مَعَ ٱللَّهِ ءَالِهَةً أُخْرَىٰ ۚ قُل لَّآ أَشْهَدُ ۚ قُلْ إِنَّمَا هُوَ إِلَهٌۭ وَحِدٌۭ وَإِنَّنِى بَرِىٓءٌۭ مِّمَّا تُشْرِكُونَ ‎</p>
 </td>
 <td width="50%">qul 'ʔay:u ʃayʔin 'ʔakbaru ʃahadatanm sl' quli 'll:ahu sl' ʃahiydunm bayniy wabaynakum j wa'ʔuwḥiya 'ʔilay:a haða' 'lqurʔa'nu li'ʔunðirakum bihiy wamanm balaɣa j 'ʔayʔin:akum lataʃhaduwna 'ʔan:a maʕa 'll:ahi ʔa'lihatan 'ʔuxray' j qul l:a'ʔ 'ʔaʃhadu j qul 'ʔin:ama' huwa 'ʔilahunm waḥidunm wa'ʔin:aniy bariyʔʔunm m:im:a' tuʃrikuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_020">20</a> ‏ٱلَّذِينَ ءَاتَيْنَهُمُ ٱلْكِتَبَ يَعْرِفُونَهُۥ كَمَا يَعْرِفُونَ أَبْنَآءَهُمُ ۘ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُمْ فَهُمْ لَا يُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">'l:aðiyna ʔa'taynahumu 'lkitaba yaʕrifuwnahuw kama' yaʕrifuwna 'ʔabna'ʔʔahumu m 'l:aðiyna xasiruwʔ' 'ʔanfusahum fahum la' yuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_021">21</a> ‏وَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ كَذِبًا أَوْ كَذَّبَ بِـَٔايَتِهِۦٓ ۗ إِنَّهُۥ لَا يُفْلِحُ ٱلظَّلِمُونَ ‎</p>
 </td>
 <td width="50%">waman 'ʔaẓlamu mim:ani 'ftaray' ʕalay 'll:ahi kaðiban' 'ʔaw kað:aba biʔa'yatihiyʔ ql' 'ʔin:ahuw la' yufliḥu 'lẓ:alimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_022">22</a> ‏وَيَوْمَ نَحْشُرُهُمْ جَمِيعًۭا ثُمَّ نَقُولُ لِلَّذِينَ أَشْرَكُوٓا۟ أَيْنَ شُرَكَآؤُكُمُ ٱلَّذِينَ كُنتُمْ تَزْعُمُونَ ‎</p>
 </td>
 <td width="50%">wayawma naḥʃuruhum jamiyʕanm' θum:a naquwlu lil:aðiyna 'ʔaʃrakuwʔ' 'ʔayna ʃuraka'ʔwʔukumu 'l:aðiyna kuntum tazʕumuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_023">23</a> ‏ثُمَّ لَمْ تَكُن فِتْنَتُهُمْ إِلَّآ أَن قَالُوا۟ وَٱللَّهِ رَبِّنَا مَا كُنَّا مُشْرِكِينَ ‎</p>
 </td>
 <td width="50%">θum:a lam takun fitnatuhum 'ʔil:a'ʔ 'ʔan qa'luw' wa'll:ahi rab:ina' ma' kun:a' muʃrikiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_024">24</a> ‏ٱنظُرْ كَيْفَ كَذَبُوا۟ عَلَىٰٓ أَنفُسِهِمْ ۚ وَضَلَّ عَنْهُم مَّا كَانُوا۟ يَفْتَرُونَ ‎</p>
 </td>
 <td width="50%">'nẓur kayfa kaðabuw' ʕalay'ʔ 'ʔanfusihim j waḍal:a ʕanhum m:a' ka'nuw' yaftaruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_006_025">25</a> ‏وَمِنْهُم مَّن يَسْتَمِعُ إِلَيْكَ ۖ وَجَعَلْنَا عَلَىٰ قُلُوبِهِمْ أَكِنَّةً أَن يَفْقَهُوهُ وَفِىٓ ءَاذَانِهِمْ وَقْرًۭا ۚ وَإِن يَرَوْا۟ كُلَّ ءَايَةٍۢ لَّا يُؤْمِنُوا۟ بِهَا ۚ حَتَّىٰٓ إِذَا جَآءُوكَ يُجَدِلُونَكَ يَقُولُ ٱلَّذِينَ كَفَرُوٓا۟ إِنْ هَذَآ إِلَّآ أَسَطِيرُ ٱلْأَوَّلِينَ ‎</p>
 </td>
 <td width="50%">waminhum m:an yastamiʕu 'ʔilayka sl' wajaʕalna' ʕalay' quluwbihim 'ʔakin:atan 'ʔan yafqahuwhu wafiyʔ ʔa'ða'nihim waqranm' j wa'ʔin yaraw' kul:a ʔa'yatinm l:a' yuwʔminuw' biha' j ḥat:ay'ʔ 'ʔiða' ja'ʔʔuwka yujadiluwnaka yaquwlu 'l:aðiyna kafaruwʔ' 'ʔin haða'ʔ 'ʔil:a'ʔ 'ʔasaṭiyru 'l'ʔaw:aliyna </td>
