 <h1 align="CENTER">76. ‏الإنسان‎</h1>
 <h2 align="CENTER">'l'ʔns'n</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_001">1</a> ‏هَلْ أَتَىٰ عَلَى ٱلْإِنسَنِ حِينٌۭ مِّنَ ٱلدَّهْرِ لَمْ يَكُن شَيْـًۭٔا مَّذْكُورًا ‎</p>
 </td>
 <td width="50%">hal 'ʔatay' ʕalay 'l'ʔinsani ḥiynunm m:ina 'ld:ahri lam yakun ʃayʔanm' m:aðkuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_002">2</a> ‏إِنَّا خَلَقْنَا ٱلْإِنسَنَ مِن نُّطْفَةٍ أَمْشَاجٍۢ نَّبْتَلِيهِ فَجَعَلْنَهُ سَمِيعًۢا بَصِيرًا ‎</p>
 </td>
 <td width="50%">'ʔin:a' xalaqna' 'l'ʔinsana min n:uṭfatin 'ʔamʃa'jinm n:abtaliyhi fajaʕalnahu samiyʕanm' baṣiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_003">3</a> ‏إِنَّا هَدَيْنَهُ ٱلسَّبِيلَ إِمَّا شَاكِرًۭا وَإِمَّا كَفُورًا ‎</p>
 </td>
 <td width="50%">'ʔin:a' hadaynahu 'ls:abiyla 'ʔim:a' ʃa'kiranm' wa'ʔim:a' kafuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_004">4</a> ‏إِنَّآ أَعْتَدْنَا لِلْكَفِرِينَ سَلَسِلَا۟ وَأَغْلَلًۭا وَسَعِيرًا ‎</p>
 </td>
 <td width="50%">'ʔin:a'ʔ 'ʔaʕtadna' lilkafiriyna salasila' wa'ʔaɣlalanm' wasaʕiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_005">5</a> ‏إِنَّ ٱلْأَبْرَارَ يَشْرَبُونَ مِن كَأْسٍۢ كَانَ مِزَاجُهَا كَافُورًا ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l'ʔabra'ra yaʃrabuwna min ka'ʔsinm ka'na miza'juha' ka'fuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_006">6</a> ‏عَيْنًۭا يَشْرَبُ بِهَا عِبَادُ ٱللَّهِ يُفَجِّرُونَهَا تَفْجِيرًۭا ‎</p>
 </td>
 <td width="50%">ʕaynanm' yaʃrabu biha' ʕiba'du 'll:ahi yufaj:iruwnaha' tafjiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_007">7</a> ‏يُوفُونَ بِٱلنَّذْرِ وَيَخَافُونَ يَوْمًۭا كَانَ شَرُّهُۥ مُسْتَطِيرًۭا ‎</p>
 </td>
 <td width="50%">yuwfuwna bi'ln:aðri wayaxa'fuwna yawmanm' ka'na ʃar:uhuw mustaṭiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_008">8</a> ‏وَيُطْعِمُونَ ٱلطَّعَامَ عَلَىٰ حُبِّهِۦ مِسْكِينًۭا وَيَتِيمًۭا وَأَسِيرًا ‎</p>
 </td>
 <td width="50%">wayuṭʕimuwna 'lṭ:aʕa'ma ʕalay' ḥub:ihiy miskiynanm' wayatiymanm' wa'ʔasiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_009">9</a> ‏إِنَّمَا نُطْعِمُكُمْ لِوَجْهِ ٱللَّهِ لَا نُرِيدُ مِنكُمْ جَزَآءًۭ وَلَا شُكُورًا ‎</p>
 </td>
 <td width="50%">'ʔin:ama' nuṭʕimukum liwajhi 'll:ahi la' nuriydu minkum jaza'ʔʔanm wala' ʃukuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_010">10</a> ‏إِنَّا نَخَافُ مِن رَّبِّنَا يَوْمًا عَبُوسًۭا قَمْطَرِيرًۭا ‎</p>
 </td>
 <td width="50%">'ʔin:a' naxa'fu min r:ab:ina' yawman' ʕabuwsanm' qamṭariyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_011">11</a> ‏فَوَقَىٰهُمُ ٱللَّهُ شَرَّ ذَلِكَ ٱلْيَوْمِ وَلَقَّىٰهُمْ نَضْرَةًۭ وَسُرُورًۭا ‎</p>
 </td>
 <td width="50%">fawaqay'humu 'll:ahu ʃar:a ðalika 'lyawmi walaq:ay'hum naḍratanm wasuruwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_012">12</a> ‏وَجَزَىٰهُم بِمَا صَبَرُوا۟ جَنَّةًۭ وَحَرِيرًۭا ‎</p>
 </td>
 <td width="50%">wajazay'hum bima' ṣabaruw' jan:atanm waḥariyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_013">13</a> ‏مُّتَّكِـِٔينَ فِيهَا عَلَى ٱلْأَرَآئِكِ ۖ لَا يَرَوْنَ فِيهَا شَمْسًۭا وَلَا زَمْهَرِيرًۭا ‎</p>
 </td>
 <td width="50%">m:ut:akiʔiyna fiyha' ʕalay 'l'ʔara'ʔyʔiki sl' la' yarawna fiyha' ʃamsanm' wala' zamhariyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_014">14</a> ‏وَدَانِيَةً عَلَيْهِمْ ظِلَلُهَا وَذُلِّلَتْ قُطُوفُهَا تَذْلِيلًۭا ‎</p>
 </td>
 <td width="50%">wada'niyatan ʕalayhim ẓilaluha' waðul:ilat quṭuwfuha' taðliylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_015">15</a> ‏وَيُطَافُ عَلَيْهِم بِـَٔانِيَةٍۢ مِّن فِضَّةٍۢ وَأَكْوَابٍۢ كَانَتْ قَوَارِيرَا۠ ‎</p>
 </td>
 <td width="50%">wayuṭa'fu ʕalayhim biʔa'niyatinm m:in fiḍ:atinm wa'ʔakwa'binm ka'nat qawa'riyra' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_016">16</a> ‏قَوَارِيرَا۟ مِن فِضَّةٍۢ قَدَّرُوهَا تَقْدِيرًۭا ‎</p>
 </td>
 <td width="50%">qawa'riyra' min fiḍ:atinm qad:aruwha' taqdiyranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_017">17</a> ‏وَيُسْقَوْنَ فِيهَا كَأْسًۭا كَانَ مِزَاجُهَا زَنجَبِيلًا ‎</p>
 </td>
 <td width="50%">wayusqawna fiyha' ka'ʔsanm' ka'na miza'juha' zanjabiylan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_018">18</a> ‏عَيْنًۭا فِيهَا تُسَمَّىٰ سَلْسَبِيلًۭا ‎</p>
 </td>
 <td width="50%">ʕaynanm' fiyha' tusam:ay' salsabiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_019">19</a> ‏ وَيَطُوفُ عَلَيْهِمْ وِلْدَنٌۭ مُّخَلَّدُونَ إِذَا رَأَيْتَهُمْ حَسِبْتَهُمْ لُؤْلُؤًۭا مَّنثُورًۭا ‎</p>
 </td>
 <td width="50%"> wayaṭuwfu ʕalayhim wildanunm m:uxal:aduwna 'ʔiða' ra'ʔaytahum ḥasibtahum luwʔluwʔanm' m:anθuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_020">20</a> ‏وَإِذَا رَأَيْتَ ثَمَّ رَأَيْتَ نَعِيمًۭا وَمُلْكًۭا كَبِيرًا ‎</p>
 </td>
 <td width="50%">wa'ʔiða' ra'ʔayta θam:a ra'ʔayta naʕiymanm' wamulkanm' kabiyran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_021">21</a> ‏عَلِيَهُمْ ثِيَابُ سُندُسٍ خُضْرٌۭ وَإِسْتَبْرَقٌۭ ۖ وَحُلُّوٓا۟ أَسَاوِرَ مِن فِضَّةٍۢ وَسَقَىٰهُمْ رَبُّهُمْ شَرَابًۭا طَهُورًا ‎</p>
 </td>
 <td width="50%">ʕaliyahum θiya'bu sundusin xuḍrunm wa'ʔistabraqunm sl' waḥul:uwʔ' 'ʔasa'wira min fiḍ:atinm wasaqay'hum rab:uhum ʃara'banm' ṭahuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_022">22</a> ‏إِنَّ هَذَا كَانَ لَكُمْ جَزَآءًۭ وَكَانَ سَعْيُكُم مَّشْكُورًا ‎</p>
 </td>
 <td width="50%">'ʔin:a haða' ka'na lakum jaza'ʔʔanm waka'na saʕyukum m:aʃkuwran' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_023">23</a> ‏إِنَّا نَحْنُ نَزَّلْنَا عَلَيْكَ ٱلْقُرْءَانَ تَنزِيلًۭا ‎</p>
 </td>
 <td width="50%">'ʔin:a' naḥnu naz:alna' ʕalayka 'lqurʔa'na tanziylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_024">24</a> ‏فَٱصْبِرْ لِحُكْمِ رَبِّكَ وَلَا تُطِعْ مِنْهُمْ ءَاثِمًا أَوْ كَفُورًۭا ‎</p>
 </td>
 <td width="50%">fa'ṣbir liḥukmi rab:ika wala' tuṭiʕ minhum ʔa'θiman' 'ʔaw kafuwranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_025">25</a> ‏وَٱذْكُرِ ٱسْمَ رَبِّكَ بُكْرَةًۭ وَأَصِيلًۭا ‎</p>
 </td>
 <td width="50%">wa'ðkuri 'sma rab:ika bukratanm wa'ʔaṣiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_026">26</a> ‏وَمِنَ ٱلَّيْلِ فَٱسْجُدْ لَهُۥ وَسَبِّحْهُ لَيْلًۭا طَوِيلًا ‎</p>
 </td>
 <td width="50%">wamina 'l:ayli fa'sjud lahuw wasab:iḥhu laylanm' ṭawiylan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_027">27</a> ‏إِنَّ هَٓؤُلَآءِ يُحِبُّونَ ٱلْعَاجِلَةَ وَيَذَرُونَ وَرَآءَهُمْ يَوْمًۭا ثَقِيلًۭا ‎</p>
 </td>
 <td width="50%">'ʔin:a haʔwʔula'ʔʔi yuḥib:uwna 'lʕa'jilata wayaðaruwna wara'ʔʔahum yawmanm' θaqiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_028">28</a> ‏نَّحْنُ خَلَقْنَهُمْ وَشَدَدْنَآ أَسْرَهُمْ ۖ وَإِذَا شِئْنَا بَدَّلْنَآ أَمْثَلَهُمْ تَبْدِيلًا ‎</p>
 </td>
 <td width="50%">n:aḥnu xalaqnahum waʃadadna'ʔ 'ʔasrahum sl' wa'ʔiða' ʃiyʔna' bad:alna'ʔ 'ʔamθalahum tabdiylan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_029">29</a> ‏إِنَّ هَذِهِۦ تَذْكِرَةٌۭ ۖ فَمَن شَآءَ ٱتَّخَذَ إِلَىٰ رَبِّهِۦ سَبِيلًۭا ‎</p>
 </td>
 <td width="50%">'ʔin:a haðihiy taðkiratunm sl' faman ʃa'ʔʔa 't:axaða 'ʔilay' rab:ihiy sabiylanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_030">30</a> ‏وَمَا تَشَآءُونَ إِلَّآ أَن يَشَآءَ ٱللَّهُ ۚ إِنَّ ٱللَّهَ كَانَ عَلِيمًا حَكِيمًۭا ‎</p>
 </td>
 <td width="50%">wama' taʃa'ʔʔuwna 'ʔil:a'ʔ 'ʔan yaʃa'ʔʔa 'll:ahu j 'ʔin:a 'll:aha ka'na ʕaliyman' ḥakiymanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_076_031">31</a> ‏يُدْخِلُ مَن يَشَآءُ فِى رَحْمَتِهِۦ ۚ وَٱلظَّلِمِينَ أَعَدَّ لَهُمْ عَذَابًا أَلِيمًۢا ‎</p>
 </td>
 <td width="50%">yudxilu man yaʃa'ʔʔu fiy raḥmatihiy j wa'lẓ:alimiyna 'ʔaʕad:a lahum ʕaða'ban' 'ʔaliymanm' </td>
 </tr>
 </table>
