 <h1 align="CENTER">64. ‏التغابن‎</h1>
 <h2 align="CENTER">'ltɣ'bn</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_001">1</a> ‏يُسَبِّحُ لِلَّهِ مَا فِى ٱلسَّمَوَتِ وَمَا فِى ٱلْأَرْضِ ۖ لَهُ ٱلْمُلْكُ وَلَهُ ٱلْحَمْدُ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ‎</p>
 </td>
 <td width="50%">yusab:iḥu lil:ahi ma' fiy 'ls:amawati wama' fiy 'l'ʔarḍi sl' lahu 'lmulku walahu 'lḥamdu sl' wahuwa ʕalay' kul:i ʃayʔinm qadiyrun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_002">2</a> ‏هُوَ ٱلَّذِى خَلَقَكُمْ فَمِنكُمْ كَافِرٌۭ وَمِنكُم مُّؤْمِنٌۭ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiy xalaqakum faminkum ka'firunm waminkum m:uwʔminunm j wa'll:ahu bima' taʕmaluwna baṣiyrun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_003">3</a> ‏خَلَقَ ٱلسَّمَوَتِ وَٱلْأَرْضَ بِٱلْحَقِّ وَصَوَّرَكُمْ فَأَحْسَنَ صُوَرَكُمْ ۖ وَإِلَيْهِ ٱلْمَصِيرُ ‎</p>
 </td>
 <td width="50%">xalaqa 'ls:amawati wa'l'ʔarḍa bi'lḥaq:i waṣaw:arakum fa'ʔaḥsana ṣuwarakum sl' wa'ʔilayhi 'lmaṣiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_004">4</a> ‏يَعْلَمُ مَا فِى ٱلسَّمَوَتِ وَٱلْأَرْضِ وَيَعْلَمُ مَا تُسِرُّونَ وَمَا تُعْلِنُونَ ۚ وَٱللَّهُ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ ‎</p>
 </td>
 <td width="50%">yaʕlamu ma' fiy 'ls:amawati wa'l'ʔarḍi wayaʕlamu ma' tusir:uwna wama' tuʕlinuwna j wa'll:ahu ʕaliymunm biða'ti 'lṣ:uduwri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_005">5</a> ‏أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ كَفَرُوا۟ مِن قَبْلُ فَذَاقُوا۟ وَبَالَ أَمْرِهِمْ وَلَهُمْ عَذَابٌ أَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">'ʔalam ya'ʔtikum nabawʔu' 'l:aðiyna kafaruw' min qablu faða'quw' waba'la 'ʔamrihim walahum ʕaða'bun 'ʔaliymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_006">6</a> ‏ذَلِكَ بِأَنَّهُۥ كَانَت تَّأْتِيهِمْ رُسُلُهُم بِٱلْبَيِّنَتِ فَقَالُوٓا۟ أَبَشَرٌۭ يَهْدُونَنَا فَكَفَرُوا۟ وَتَوَلَّوا۟ ۚ وَّٱسْتَغْنَى ٱللَّهُ ۚ وَٱللَّهُ غَنِىٌّ حَمِيدٌۭ ‎</p>
 </td>
 <td width="50%">ðalika bi'ʔan:ahuw ka'nat t:a'ʔtiyhim rusuluhum bi'lbay:inati faqa'luwʔ' 'ʔabaʃarunm yahduwnana' fakafaruw' watawal:aw' j w:a'staɣnay 'll:ahu j wa'll:ahu ɣaniy:un ḥamiydunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_007">7</a> ‏زَعَمَ ٱلَّذِينَ كَفَرُوٓا۟ أَن لَّن يُبْعَثُوا۟ ۚ قُلْ بَلَىٰ وَرَبِّى لَتُبْعَثُنَّ ثُمَّ لَتُنَبَّؤُنَّ بِمَا عَمِلْتُمْ ۚ وَذَلِكَ عَلَى ٱللَّهِ يَسِيرٌۭ ‎</p>
 </td>
 <td width="50%">zaʕama 'l:aðiyna kafaruwʔ' 'ʔan l:an yubʕaθuw' j qul balay' warab:iy latubʕaθun:a θum:a latunab:awʔun:a bima' ʕamiltum j waðalika ʕalay 'll:ahi yasiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_008">8</a> ‏فَـَٔامِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَٱلنُّورِ ٱلَّذِىٓ أَنزَلْنَا ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ ‎</p>
 </td>
 <td width="50%">faʔa'minuw' bi'll:ahi warasuwlihiy wa'ln:uwri 'l:aðiyʔ 'ʔanzalna' j wa'll:ahu bima' taʕmaluwna xabiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_009">9</a> ‏يَوْمَ يَجْمَعُكُمْ لِيَوْمِ ٱلْجَمْعِ ۖ ذَلِكَ يَوْمُ ٱلتَّغَابُنِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ وَيَعْمَلْ صَلِحًۭا يُكَفِّرْ عَنْهُ سَيِّـَٔاتِهِۦ وَيُدْخِلْهُ جَنَّتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَرُ خَلِدِينَ فِيهَآ أَبَدًۭا ۚ ذَلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ‎</p>
 </td>
 <td width="50%">yawma yajmaʕukum liyawmi 'ljamʕi sl' ðalika yawmu 'lt:aɣa'buni ql' waman yuwʔminm bi'll:ahi wayaʕmal ṣaliḥanm' yukaf:ir ʕanhu say:iʔa'tihiy wayudxilhu jan:atinm tajriy min taḥtiha' 'l'ʔanharu xalidiyna fiyha'ʔ 'ʔabadanm' j ðalika 'lfawzu 'lʕaẓiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_010">10</a> ‏وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَتِنَآ أُو۟لَٓئِكَ أَصْحَبُ ٱلنَّارِ خَلِدِينَ فِيهَا ۖ وَبِئْسَ ٱلْمَصِيرُ ‎</p>
 </td>
 <td width="50%">wa'l:aðiyna kafaruw' wakað:abuw' biʔa'yatina'ʔ 'ʔuwlaʔyʔika 'ʔaṣḥabu 'ln:a'ri xalidiyna fiyha' sl' wabiyʔsa 'lmaṣiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_011">11</a> ‏مَآ أَصَابَ مِن مُّصِيبَةٍ إِلَّا بِإِذْنِ ٱللَّهِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ يَهْدِ قَلْبَهُۥ ۚ وَٱللَّهُ بِكُلِّ شَىْءٍ عَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">ma'ʔ 'ʔaṣa'ba min m:uṣiybatin 'ʔil:a' bi'ʔiðni 'll:ahi ql' waman yuwʔminm bi'll:ahi yahdi qalbahuw j wa'll:ahu bikul:i ʃayʔin ʕaliymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_012">12</a> ‏وَأَطِيعُوا۟ ٱللَّهَ وَأَطِيعُوا۟ ٱلرَّسُولَ ۚ فَإِن تَوَلَّيْتُمْ فَإِنَّمَا عَلَىٰ رَسُولِنَا ٱلْبَلَغُ ٱلْمُبِينُ ‎</p>
 </td>
 <td width="50%">wa'ʔaṭiyʕuw' 'll:aha wa'ʔaṭiyʕuw' 'lr:asuwla j fa'ʔin tawal:aytum fa'ʔin:ama' ʕalay' rasuwlina' 'lbalaɣu 'lmubiynu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_013">13</a> ‏ٱللَّهُ لَآ إِلَهَ إِلَّا هُوَ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">'ll:ahu la'ʔ 'ʔilaha 'ʔil:a' huwa j waʕalay 'll:ahi falyatawak:ali 'lmuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_014">14</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّ مِنْ أَزْوَجِكُمْ وَأَوْلَدِكُمْ عَدُوًّۭا لَّكُمْ فَٱحْذَرُوهُمْ ۚ وَإِن تَعْفُوا۟ وَتَصْفَحُوا۟ وَتَغْفِرُوا۟ فَإِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuwʔ' 'ʔin:a min 'ʔazwajikum wa'ʔawladikum ʕaduw:anm' l:akum fa'ḥðaruwhum j wa'ʔin taʕfuw' wataṣfaḥuw' wataɣfiruw' fa'ʔin:a 'll:aha ɣafuwrunm r:aḥiymun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_015">15</a> ‏إِنَّمَآ أَمْوَلُكُمْ وَأَوْلَدُكُمْ فِتْنَةٌۭ ۚ وَٱللَّهُ عِندَهُۥٓ أَجْرٌ عَظِيمٌۭ ‎</p>
 </td>
 <td width="50%">'ʔin:ama'ʔ 'ʔamwalukum wa'ʔawladukum fitnatunm j wa'll:ahu ʕindahuwʔ 'ʔajrun ʕaẓiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_016">16</a> ‏فَٱتَّقُوا۟ ٱللَّهَ مَا ٱسْتَطَعْتُمْ وَٱسْمَعُوا۟ وَأَطِيعُوا۟ وَأَنفِقُوا۟ خَيْرًۭا لِأَنفُسِكُمْ ۗ وَمَن يُوقَ شُحَّ نَفْسِهِۦ فَأُو۟لَٓئِكَ هُمُ ٱلْمُفْلِحُونَ ‎</p>
 </td>
 <td width="50%">fa't:aquw' 'll:aha ma' 'staṭaʕtum wa'smaʕuw' wa'ʔaṭiyʕuw' wa'ʔanfiquw' xayranm' li'ʔanfusikum ql' waman yuwqa ʃuḥ:a nafsihiy fa'ʔuwlaʔyʔika humu 'lmufliḥuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_017">17</a> ‏إِن تُقْرِضُوا۟ ٱللَّهَ قَرْضًا حَسَنًۭا يُضَعِفْهُ لَكُمْ وَيَغْفِرْ لَكُمْ ۚ وَٱللَّهُ شَكُورٌ حَلِيمٌ ‎</p>
 </td>
 <td width="50%">'ʔin tuqriḍuw' 'll:aha qarḍan' ḥasananm' yuḍaʕifhu lakum wayaɣfir lakum j wa'll:ahu ʃakuwrun ḥaliymun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_064_018">18</a> ‏عَلِمُ ٱلْغَيْبِ وَٱلشَّهَدَةِ ٱلْعَزِيزُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">ʕalimu 'lɣaybi wa'lʃ:ahadati 'lʕaziyzu 'lḥakiymu </td>
 </tr>
 </table>
