 <h1 align="CENTER">26. ‏الشعراء‎</h1>
 <h2 align="CENTER">'lʃʕr'ʔ</h2>

 </td>

 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_003">3</a> ‏لَعَلَّكَ بَخِعٌۭ نَّفْسَكَ أَلَّا يَكُونُوا۟ مُؤْمِنِينَ ‎</p>
 </td>
 <td width="50%">laʕal:aka baxiʕunm n:afsaka 'ʔal:a' yakuwnuw' muwʔminiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_004">4</a> ‏إِن نَّشَأْ نُنَزِّلْ عَلَيْهِم مِّنَ ٱلسَّمَآءِ ءَايَةًۭ فَظَلَّتْ أَعْنَقُهُمْ لَهَا خَضِعِينَ ‎</p>
 </td>
 <td width="50%">'ʔin n:aʃa'ʔ nunaz:il ʕalayhim m:ina 'ls:ama'ʔʔi ʔa'yatanm faẓal:at 'ʔaʕnaquhum laha' xaḍiʕiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_005">5</a> ‏وَمَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّنَ ٱلرَّحْمَنِ مُحْدَثٍ إِلَّا كَانُوا۟ عَنْهُ مُعْرِضِينَ ‎</p>
 </td>
 <td width="50%">wama' ya'ʔtiyhim m:in ðikrinm m:ina 'lr:aḥmani muḥdaθin 'ʔil:a' ka'nuw' ʕanhu muʕriḍiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_006">6</a> ‏فَقَدْ كَذَّبُوا۟ فَسَيَأْتِيهِمْ أَنۢبَٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ ‎</p>
 </td>
 <td width="50%">faqad kað:abuw' fasaya'ʔtiyhim 'ʔanmbaʔwʔu' ma' ka'nuw' bihiy yastahziʔuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_007">7</a> ‏أَوَلَمْ يَرَوْا۟ إِلَى ٱلْأَرْضِ كَمْ أَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٍۢ كَرِيمٍ ‎</p>
 </td>
 <td width="50%">'ʔawalam yaraw' 'ʔilay 'l'ʔarḍi kam 'ʔanmbatna' fiyha' min kul:i zawjinm kariymin </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_008">8</a> ‏إِنَّ فِى ذَلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ ‎</p>
 </td>
 <td width="50%">'ʔin:a fiy ðalika laʔa'yatanm sl' wama' ka'na 'ʔakθaruhum m:uwʔminiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_009">9</a> ‏وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ ‎</p>
 </td>

 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_010">10</a> ‏وَإِذْ نَادَىٰ رَبُّكَ مُوسَىٰٓ أَنِ ٱئْتِ ٱلْقَوْمَ ٱلظَّلِمِينَ ‎</p>
 </td>

 <td width="50%"><p align="RIGHT"><a name="an_026_011">11</a> ‏قَوْمَ فِرْعَوْنَ ۚ أَلَا يَتَّقُونَ ‎</p>

 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_012">12</a> ‏قَالَ رَبِّ إِنِّىٓ أَخَافُ أَن يُكَذِّبُونِ ‎</p>
 </td>
 <td width="50%">qa'la rab:i 'ʔin:iyʔ 'ʔaxa'fu 'ʔan yukað:ibuwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_013">13</a> ‏وَيَضِيقُ صَدْرِى وَلَا يَنطَلِقُ لِسَانِى فَأَرْسِلْ إِلَىٰ هَرُونَ ‎</p>
 </td>
 <td width="50%">wayaḍiyqu ṣadriy wala' yanṭaliqu lisa'niy fa'ʔarsil 'ʔilay' haruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_014">14</a> ‏وَلَهُمْ عَلَىَّ ذَنۢبٌۭ فَأَخَافُ أَن يَقْتُلُونِ ‎</p>
 </td>
 <td width="50%">walahum ʕalay:a ðanmbunm fa'ʔaxa'fu 'ʔan yaqtuluwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_015">15</a> ‏قَالَ كَلَّا ۖ فَٱذْهَبَا بِـَٔايَتِنَآ ۖ إِنَّا مَعَكُم مُّسْتَمِعُونَ ‎</p>
 </td>
 <td width="50%">qa'la kal:a' sl' fa'ðhaba' biʔa'yatina'ʔ sl' 'ʔin:a' maʕakum m:ustamiʕuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_016">16</a> ‏فَأْتِيَا فِرْعَوْنَ فَقُولَآ إِنَّا رَسُولُ رَبِّ ٱلْعَلَمِينَ ‎</p>
 </td>

 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_017">17</a> ‏أَنْ أَرْسِلْ مَعَنَا بَنِىٓ إِسْرَٓءِيلَ ‎</p>

 <td width="50%">'ʔan 'ʔarsil maʕana' baniyʔ 'ʔisraʔʔiyla </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_018">18</a> ‏قَالَ أَلَمْ نُرَبِّكَ فِينَا وَلِيدًۭا وَلَبِثْتَ فِينَا مِنْ عُمُرِكَ سِنِينَ ‎</p>

 <td width="50%">qa'la 'ʔalam nurab:ika fiyna' waliydanm' walabiθta fiyna' min ʕumurika siniyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_019">19</a> ‏وَفَعَلْتَ فَعْلَتَكَ ٱلَّتِى فَعَلْتَ وَأَنتَ مِنَ ٱلْكَفِرِينَ ‎</p>

 <td width="50%">wafaʕalta faʕlataka 'l:atiy faʕalta wa'ʔanta mina 'lkafiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_020">20</a> ‏قَالَ فَعَلْتُهَآ إِذًۭا وَأَنَا۠ مِنَ ٱلضَّآلِّينَ ‎</p>
 </td>
 <td width="50%">qa'la faʕaltuha'ʔ 'ʔiðanm' wa'ʔana' mina 'lḍ:a'ʔl:iyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_021">21</a> ‏فَفَرَرْتُ مِنكُمْ لَمَّا خِفْتُكُمْ فَوَهَبَ لِى رَبِّى حُكْمًۭا وَجَعَلَنِى مِنَ ٱلْمُرْسَلِينَ ‎</p>
 </td>
 <td width="50%">fafarartu minkum lam:a' xiftukum fawahaba liy rab:iy ḥukmanm' wajaʕalaniy mina 'lmursaliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_022">22</a> ‏وَتِلْكَ نِعْمَةٌۭ تَمُنُّهَا عَلَىَّ أَنْ عَبَّدتَّ بَنِىٓ إِسْرَٓءِيلَ ‎</p>
 </td>
 <td width="50%">watilka niʕmatunm tamun:uha' ʕalay:a 'ʔan ʕab:adt:a baniyʔ 'ʔisraʔʔiyla </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_023">23</a> ‏قَالَ فِرْعَوْنُ وَمَا رَبُّ ٱلْعَلَمِينَ ‎</p>
 </td>
 <td width="50%">qa'la firʕawnu wama' rab:u 'lʕalamiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_024">24</a> ‏قَالَ رَبُّ ٱلسَّمَوَتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُم مُّوقِنِينَ ‎</p>
 </td>
 <td width="50%">qa'la rab:u 'ls:amawati wa'l'ʔarḍi wama' baynahuma'ʔ sl' 'ʔin kuntum m:uwqiniyna </td>

 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_025">25</a> ‏قَالَ لِمَنْ حَوْلَهُۥٓ أَلَا تَسْتَمِعُونَ ‎</p>
 </td>
 <td width="50%">qa'la liman ḥawlahuwʔ 'ʔala' tastamiʕuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_026">26</a> ‏قَالَ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ ‎</p>
 </td>
 <td width="50%">qa'la rab:ukum warab:u ʔa'ba'ʔyʔikumu 'l'ʔaw:aliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_027">27</a> ‏قَالَ إِنَّ رَسُولَكُمُ ٱلَّذِىٓ أُرْسِلَ إِلَيْكُمْ لَمَجْنُونٌۭ ‎</p>
 </td>
 <td width="50%">qa'la 'ʔin:a rasuwlakumu 'l:aðiyʔ 'ʔursila 'ʔilaykum lamajnuwnunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_028">28</a> ‏قَالَ رَبُّ ٱلْمَشْرِقِ وَٱلْمَغْرِبِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُمْ تَعْقِلُونَ ‎</p>
 </td>
 <td width="50%">qa'la rab:u 'lmaʃriqi wa'lmaɣribi wama' baynahuma'ʔ sl' 'ʔin kuntum taʕqiluwna </td>

 <td width="50%"><p align="RIGHT"><a name="an_026_029">29</a> ‏قَالَ لَىِٕنِ ٱتَّخَذْتَ إِلَهًا غَيْرِى لَأَجْعَلَنَّكَ مِنَ ٱلْمَسْجُونِينَ ‎</p>

 <td width="50%"><p align="RIGHT"><a name="an_026_030">30</a> ‏قَالَ أَوَلَوْ جِئْتُكَ بِشَىْءٍۢ مُّبِينٍۢ ‎</p>

 <td width="50%">qa'la fa'ʔti bihiyʔ 'ʔin kunta mina 'lṣ:adiqiyna </td>
 </tr>

 </td>
 <td width="50%">fa'ʔalqay' ʕaṣa'hu fa'ʔiða' hiya θuʕba'nunm m:ubiynunm </td>

 </td>
 <td width="50%">wanazaʕa yadahuw fa'ʔiða' hiya bayḍa'ʔʔu liln:aẓiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_034">34</a> ‏قَالَ لِلْمَلَإِ حَوْلَهُۥٓ إِنَّ هَذَا لَسَحِرٌ عَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">qa'la lilmala'ʔi ḥawlahuwʔ 'ʔin:a haða' lasaḥirun ʕaliymunm </td>
 </tr>

 </tr>

 <td width="50%"><p align="RIGHT"><a name="an_026_036">36</a> ‏قَالُوٓا۟ أَرْجِهْ وَأَخَاهُ وَٱبْعَثْ فِى ٱلْمَدَآىِٕنِ حَشِرِينَ ‎</p>
 </td>
 <td width="50%">qa'luwʔ' 'ʔarjih wa'ʔaxa'hu wa'bʕaθ fiy 'lmada'ʔyʔini ḥaʃiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_037">37</a> ‏يَأْتُوكَ بِكُلِّ سَحَّارٍ عَلِيمٍۢ ‎</p>
 </td>
 <td width="50%">ya'ʔtuwka bikul:i saḥ:a'rin ʕaliyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_038">38</a> ‏فَجُمِعَ ٱلسَّحَرَةُ لِمِيقَتِ يَوْمٍۢ مَّعْلُومٍۢ ‎</p>
 </td>
 <td width="50%">fajumiʕa 'ls:aḥaratu limiyqati yawminm m:aʕluwminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_039">39</a> ‏وَقِيلَ لِلنَّاسِ هَلْ أَنتُم مُّجْتَمِعُونَ ‎</p>
 </td>
 <td width="50%">waqiyla liln:a'si hal 'ʔantum m:ujtamiʕuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_040">40</a> ‏لَعَلَّنَا نَتَّبِعُ ٱلسَّحَرَةَ إِن كَانُوا۟ هُمُ ٱلْغَلِبِينَ ‎</p>
 </td>
 <td width="50%">laʕal:ana' nat:abiʕu 'ls:aḥarata 'ʔin ka'nuw' humu 'lɣalibiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_041">41</a> ‏فَلَمَّا جَآءَ ٱلسَّحَرَةُ قَالُوا۟ لِفِرْعَوْنَ أَىِٕنَّ لَنَا لَأَجْرًا إِن كُنَّا نَحْنُ ٱلْغَلِبِينَ ‎</p>
 </td>
 <td width="50%">falam:a' ja'ʔʔa 'ls:aḥaratu qa'luw' lifirʕawna 'ʔayʔin:a lana' la'ʔajran' 'ʔin kun:a' naḥnu 'lɣalibiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_042">42</a> ‏قَالَ نَعَمْ وَإِنَّكُمْ إِذًۭا لَّمِنَ ٱلْمُقَرَّبِينَ ‎</p>
 </td>
 <td width="50%">qa'la naʕam wa'ʔin:akum 'ʔiðanm' l:amina 'lmuqar:abiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_026_043">43</a> ‏قَالَ لَهُم مُّوسَىٰٓ أَلْقُوا۟ مَآ أَنتُم مُّلْقُونَ ‎</p>
 </td>
 <td width="50%">qa'la lahum m:uwsay'ʔ 'ʔalquw' ma'ʔ 'ʔantum m:ulquwna </td>
 </tr>
 <tr valign="TOP">

 </td>
 <td width="50%">fa'ʔalqaw' ḥiba'lahum waʕiṣiy:ahum waqa'luw' biʕiz:ati firʕawna 'ʔin:a' lanaḥnu 'lɣalibuwna </td>
 </tr>
 <tr valign="TOP">
