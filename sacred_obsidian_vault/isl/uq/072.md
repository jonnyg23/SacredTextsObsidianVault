 <h1 align="CENTER">72. ‏الجن‎</h1>
 <h2 align="CENTER">'ljn</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_001">1</a> ‏قُلْ أُوحِىَ إِلَىَّ أَنَّهُ ٱسْتَمَعَ نَفَرٌۭ مِّنَ ٱلْجِنِّ فَقَالُوٓا۟ إِنَّا سَمِعْنَا قُرْءَانًا عَجَبًۭا ‎</p>
 </td>
 <td width="50%">qul 'ʔuwḥiya 'ʔilay:a 'ʔan:ahu 'stamaʕa nafarunm m:ina 'ljin:i faqa'luwʔ' 'ʔin:a' samiʕna' qurʔa'nan' ʕajabanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_002">2</a> ‏يَهْدِىٓ إِلَى ٱلرُّشْدِ فَـَٔامَنَّا بِهِۦ ۖ وَلَن نُّشْرِكَ بِرَبِّنَآ أَحَدًۭا ‎</p>
 </td>
 <td width="50%">yahdiyʔ 'ʔilay 'lr:uʃdi faʔa'man:a' bihiy sl' walan n:uʃrika birab:ina'ʔ 'ʔaḥadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_003">3</a> ‏وَأَنَّهُۥ تَعَلَىٰ جَدُّ رَبِّنَا مَا ٱتَّخَذَ صَحِبَةًۭ وَلَا وَلَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:ahuw taʕalay' jad:u rab:ina' ma' 't:axaða ṣaḥibatanm wala' waladanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_004">4</a> ‏وَأَنَّهُۥ كَانَ يَقُولُ سَفِيهُنَا عَلَى ٱللَّهِ شَطَطًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:ahuw ka'na yaquwlu safiyhuna' ʕalay 'll:ahi ʃaṭaṭanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_005">5</a> ‏وَأَنَّا ظَنَنَّآ أَن لَّن تَقُولَ ٱلْإِنسُ وَٱلْجِنُّ عَلَى ٱللَّهِ كَذِبًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' ẓanan:a'ʔ 'ʔan l:an taquwla 'l'ʔinsu wa'ljin:u ʕalay 'll:ahi kaðibanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_006">6</a> ‏وَأَنَّهُۥ كَانَ رِجَالٌۭ مِّنَ ٱلْإِنسِ يَعُوذُونَ بِرِجَالٍۢ مِّنَ ٱلْجِنِّ فَزَادُوهُمْ رَهَقًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:ahuw ka'na rija'lunm m:ina 'l'ʔinsi yaʕuwðuwna birija'linm m:ina 'ljin:i faza'duwhum rahaqanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_007">7</a> ‏وَأَنَّهُمْ ظَنُّوا۟ كَمَا ظَنَنتُمْ أَن لَّن يَبْعَثَ ٱللَّهُ أَحَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:ahum ẓan:uw' kama' ẓanantum 'ʔan l:an yabʕaθa 'll:ahu 'ʔaḥadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_008">8</a> ‏وَأَنَّا لَمَسْنَا ٱلسَّمَآءَ فَوَجَدْنَهَا مُلِئَتْ حَرَسًۭا شَدِيدًۭا وَشُهُبًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' lamasna' 'ls:ama'ʔʔa fawajadnaha' muliyʔat ḥarasanm' ʃadiydanm' waʃuhubanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_009">9</a> ‏وَأَنَّا كُنَّا نَقْعُدُ مِنْهَا مَقَعِدَ لِلسَّمْعِ ۖ فَمَن يَسْتَمِعِ ٱلْءَانَ يَجِدْ لَهُۥ شِهَابًۭا رَّصَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' kun:a' naqʕudu minha' maqaʕida lils:amʕi sl' faman yastamiʕi 'lʔa'na yajid lahuw ʃiha'banm' r:aṣadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_010">10</a> ‏وَأَنَّا لَا نَدْرِىٓ أَشَرٌّ أُرِيدَ بِمَن فِى ٱلْأَرْضِ أَمْ أَرَادَ بِهِمْ رَبُّهُمْ رَشَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' la' nadriyʔ 'ʔaʃar:un 'ʔuriyda biman fiy 'l'ʔarḍi 'ʔam 'ʔara'da bihim rab:uhum raʃadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_011">11</a> ‏وَأَنَّا مِنَّا ٱلصَّلِحُونَ وَمِنَّا دُونَ ذَلِكَ ۖ كُنَّا طَرَآئِقَ قِدَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' min:a' 'lṣ:aliḥuwna wamin:a' duwna ðalika sl' kun:a' ṭara'ʔyʔiqa qidadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_012">12</a> ‏وَأَنَّا ظَنَنَّآ أَن لَّن نُّعْجِزَ ٱللَّهَ فِى ٱلْأَرْضِ وَلَن نُّعْجِزَهُۥ هَرَبًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' ẓanan:a'ʔ 'ʔan l:an n:uʕjiza 'll:aha fiy 'l'ʔarḍi walan n:uʕjizahuw harabanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_013">13</a> ‏وَأَنَّا لَمَّا سَمِعْنَا ٱلْهُدَىٰٓ ءَامَنَّا بِهِۦ ۖ فَمَن يُؤْمِنۢ بِرَبِّهِۦ فَلَا يَخَافُ بَخْسًۭا وَلَا رَهَقًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' lam:a' samiʕna' 'lhuday'ʔ ʔa'man:a' bihiy sl' faman yuwʔminm birab:ihiy fala' yaxa'fu baxsanm' wala' rahaqanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_014">14</a> ‏وَأَنَّا مِنَّا ٱلْمُسْلِمُونَ وَمِنَّا ٱلْقَسِطُونَ ۖ فَمَنْ أَسْلَمَ فَأُو۟لَٓئِكَ تَحَرَّوْا۟ رَشَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a' min:a' 'lmuslimuwna wamin:a' 'lqasiṭuwna sl' faman 'ʔaslama fa'ʔuwlaʔyʔika taḥar:aw' raʃadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_015">15</a> ‏وَأَمَّا ٱلْقَسِطُونَ فَكَانُوا۟ لِجَهَنَّمَ حَطَبًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔam:a' 'lqasiṭuwna faka'nuw' lijahan:ama ḥaṭabanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_016">16</a> ‏وَأَلَّوِ ٱسْتَقَمُوا۟ عَلَى ٱلطَّرِيقَةِ لَأَسْقَيْنَهُم مَّآءً غَدَقًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔal:awi 'staqamuw' ʕalay 'lṭ:ariyqati la'ʔasqaynahum m:a'ʔʔan ɣadaqanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_017">17</a> ‏لِّنَفْتِنَهُمْ فِيهِ ۚ وَمَن يُعْرِضْ عَن ذِكْرِ رَبِّهِۦ يَسْلُكْهُ عَذَابًۭا صَعَدًۭا ‎</p>
 </td>
 <td width="50%">l:inaftinahum fiyhi j waman yuʕriḍ ʕan ðikri rab:ihiy yaslukhu ʕaða'banm' ṣaʕadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_018">18</a> ‏وَأَنَّ ٱلْمَسَجِدَ لِلَّهِ فَلَا تَدْعُوا۟ مَعَ ٱللَّهِ أَحَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:a 'lmasajida lil:ahi fala' tadʕuw' maʕa 'll:ahi 'ʔaḥadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_019">19</a> ‏وَأَنَّهُۥ لَمَّا قَامَ عَبْدُ ٱللَّهِ يَدْعُوهُ كَادُوا۟ يَكُونُونَ عَلَيْهِ لِبَدًۭا ‎</p>
 </td>
 <td width="50%">wa'ʔan:ahuw lam:a' qa'ma ʕabdu 'll:ahi yadʕuwhu ka'duw' yakuwnuwna ʕalayhi libadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_020">20</a> ‏قُلْ إِنَّمَآ أَدْعُوا۟ رَبِّى وَلَآ أُشْرِكُ بِهِۦٓ أَحَدًۭا ‎</p>
 </td>
 <td width="50%">qul 'ʔin:ama'ʔ 'ʔadʕuw' rab:iy wala'ʔ 'ʔuʃriku bihiyʔ 'ʔaḥadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_021">21</a> ‏قُلْ إِنِّى لَآ أَمْلِكُ لَكُمْ ضَرًّۭا وَلَا رَشَدًۭا ‎</p>
 </td>
 <td width="50%">qul 'ʔin:iy la'ʔ 'ʔamliku lakum ḍar:anm' wala' raʃadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_022">22</a> ‏قُلْ إِنِّى لَن يُجِيرَنِى مِنَ ٱللَّهِ أَحَدٌۭ وَلَنْ أَجِدَ مِن دُونِهِۦ مُلْتَحَدًا ‎</p>
 </td>
 <td width="50%">qul 'ʔin:iy lan yujiyraniy mina 'll:ahi 'ʔaḥadunm walan 'ʔajida min duwnihiy multaḥadan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_023">23</a> ‏إِلَّا بَلَغًۭا مِّنَ ٱللَّهِ وَرِسَلَتِهِۦ ۚ وَمَن يَعْصِ ٱللَّهَ وَرَسُولَهُۥ فَإِنَّ لَهُۥ نَارَ جَهَنَّمَ خَلِدِينَ فِيهَآ أَبَدًا ‎</p>
 </td>
 <td width="50%">'ʔil:a' balaɣanm' m:ina 'll:ahi warisalatihiy j waman yaʕṣi 'll:aha warasuwlahuw fa'ʔin:a lahuw na'ra jahan:ama xalidiyna fiyha'ʔ 'ʔabadan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_024">24</a> ‏حَتَّىٰٓ إِذَا رَأَوْا۟ مَا يُوعَدُونَ فَسَيَعْلَمُونَ مَنْ أَضْعَفُ نَاصِرًۭا وَأَقَلُّ عَدَدًۭا ‎</p>
 </td>
 <td width="50%">ḥat:ay'ʔ 'ʔiða' ra'ʔaw' ma' yuwʕaduwna fasayaʕlamuwna man 'ʔaḍʕafu na'ṣiranm' wa'ʔaqal:u ʕadadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_025">25</a> ‏قُلْ إِنْ أَدْرِىٓ أَقَرِيبٌۭ مَّا تُوعَدُونَ أَمْ يَجْعَلُ لَهُۥ رَبِّىٓ أَمَدًا ‎</p>
 </td>
 <td width="50%">qul 'ʔin 'ʔadriyʔ 'ʔaqariybunm m:a' tuwʕaduwna 'ʔam yajʕalu lahuw rab:iyʔ 'ʔamadan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_026">26</a> ‏عَلِمُ ٱلْغَيْبِ فَلَا يُظْهِرُ عَلَىٰ غَيْبِهِۦٓ أَحَدًا ‎</p>
 </td>
 <td width="50%">ʕalimu 'lɣaybi fala' yuẓhiru ʕalay' ɣaybihiyʔ 'ʔaḥadan' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_027">27</a> ‏إِلَّا مَنِ ٱرْتَضَىٰ مِن رَّسُولٍۢ فَإِنَّهُۥ يَسْلُكُ مِنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦ رَصَدًۭا ‎</p>
 </td>
 <td width="50%">'ʔil:a' mani 'rtaḍay' min r:asuwlinm fa'ʔin:ahuw yasluku minm bayni yadayhi wamin xalfihiy raṣadanm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_072_028">28</a> ‏لِّيَعْلَمَ أَن قَدْ أَبْلَغُوا۟ رِسَلَتِ رَبِّهِمْ وَأَحَاطَ بِمَا لَدَيْهِمْ وَأَحْصَىٰ كُلَّ شَىْءٍ عَدَدًۢا ‎</p>
 </td>
 <td width="50%">l:iyaʕlama 'ʔan qad 'ʔablaɣuw' risalati rab:ihim wa'ʔaḥa'ṭa bima' ladayhim wa'ʔaḥṣay' kul:a ʃayʔin ʕadadanm' </td>
 </tr>
