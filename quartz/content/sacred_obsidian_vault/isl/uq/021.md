 <h1 align="CENTER">21. ‏الأنبياء‎</h1>
 <h2 align="CENTER">'l'ʔnby'ʔ</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_001">1</a> ‏ٱقْتَرَبَ لِلنَّاسِ حِسَابُهُمْ وَهُمْ فِى غَفْلَةٍۢ مُّعْرِضُونَ ‎</p>
 </td>
 <td width="50%">'qtaraba liln:a'si ḥisa'buhum wahum fiy ɣaflatinm m:uʕriḍuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_002">2</a> ‏مَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّن رَّبِّهِم مُّحْدَثٍ إِلَّا ٱسْتَمَعُوهُ وَهُمْ يَلْعَبُونَ ‎</p>
 </td>
 <td width="50%">ma' ya'ʔtiyhim m:in ðikrinm m:in r:ab:ihim m:uḥdaθin 'ʔil:a' 'stamaʕuwhu wahum yalʕabuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_003">3</a> ‏لَاهِيَةًۭ قُلُوبُهُمْ ۗ وَأَسَرُّوا۟ ٱلنَّجْوَى ٱلَّذِينَ ظَلَمُوا۟ هَلْ هَذَآ إِلَّا بَشَرٌۭ مِّثْلُكُمْ ۖ أَفَتَأْتُونَ ٱلسِّحْرَ وَأَنتُمْ تُبْصِرُونَ ‎</p>
 </td>
 <td width="50%">la'hiyatanm quluwbuhum ql' wa'ʔasar:uw' 'ln:ajway 'l:aðiyna ẓalamuw' hal haða'ʔ 'ʔil:a' baʃarunm m:iθlukum sl' 'ʔafata'ʔtuwna 'ls:iḥra wa'ʔantum tubṣiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_004">4</a> ‏قَالَ رَبِّى يَعْلَمُ ٱلْقَوْلَ فِى ٱلسَّمَآءِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ ‎</p>
 </td>
 <td width="50%">qa'la rab:iy yaʕlamu 'lqawla fiy 'ls:ama'ʔʔi wa'l'ʔarḍi sl' wahuwa 'ls:amiyʕu 'lʕaliymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_005">5</a> ‏بَلْ قَالُوٓا۟ أَضْغَثُ أَحْلَمٍۭ بَلِ ٱفْتَرَىٰهُ بَلْ هُوَ شَاعِرٌۭ فَلْيَأْتِنَا بِـَٔايَةٍۢ كَمَآ أُرْسِلَ ٱلْأَوَّلُونَ ‎</p>
 </td>
 <td width="50%">bal qa'luwʔ' 'ʔaḍɣaθu 'ʔaḥlaminm bali 'ftaray'hu bal huwa ʃa'ʕirunm falya'ʔtina' biʔa'yatinm kama'ʔ 'ʔursila 'l'ʔaw:aluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_006">6</a> ‏مَآ ءَامَنَتْ قَبْلَهُم مِّن قَرْيَةٍ أَهْلَكْنَهَآ ۖ أَفَهُمْ يُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">ma'ʔ ʔa'manat qablahum m:in qaryatin 'ʔahlaknaha'ʔ sl' 'ʔafahum yuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_007">7</a> ‏وَمَآ أَرْسَلْنَا قَبْلَكَ إِلَّا رِجَالًۭا نُّوحِىٓ إِلَيْهِمْ ۖ فَسْـَٔلُوٓا۟ أَهْلَ ٱلذِّكْرِ إِن كُنتُمْ لَا تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">wama'ʔ 'ʔarsalna' qablaka 'ʔil:a' rija'lanm' n:uwḥiyʔ 'ʔilayhim sl' fasʔaluwʔ' 'ʔahla 'lð:ikri 'ʔin kuntum la' taʕlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_008">8</a> ‏وَمَا جَعَلْنَهُمْ جَسَدًۭا لَّا يَأْكُلُونَ ٱلطَّعَامَ وَمَا كَانُوا۟ خَلِدِينَ ‎</p>
 </td>
 <td width="50%">wama' jaʕalnahum jasadanm' l:a' ya'ʔkuluwna 'lṭ:aʕa'ma wama' ka'nuw' xalidiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_009">9</a> ‏ثُمَّ صَدَقْنَهُمُ ٱلْوَعْدَ فَأَنجَيْنَهُمْ وَمَن نَّشَآءُ وَأَهْلَكْنَا ٱلْمُسْرِفِينَ ‎</p>
 </td>
 <td width="50%">θum:a ṣadaqnahumu 'lwaʕda fa'ʔanjaynahum waman n:aʃa'ʔʔu wa'ʔahlakna' 'lmusrifiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_010">10</a> ‏لَقَدْ أَنزَلْنَآ إِلَيْكُمْ كِتَبًۭا فِيهِ ذِكْرُكُمْ ۖ أَفَلَا تَعْقِلُونَ ‎</p>
 </td>
 <td width="50%">laqad 'ʔanzalna'ʔ 'ʔilaykum kitabanm' fiyhi ðikrukum sl' 'ʔafala' taʕqiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_011">11</a> ‏وَكَمْ قَصَمْنَا مِن قَرْيَةٍۢ كَانَتْ ظَالِمَةًۭ وَأَنشَأْنَا بَعْدَهَا قَوْمًا ءَاخَرِينَ ‎</p>
 </td>
 <td width="50%">wakam qaṣamna' min qaryatinm ka'nat ẓa'limatanm wa'ʔanʃa'ʔna' baʕdaha' qawman' ʔa'xariyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_012">12</a> ‏فَلَمَّآ أَحَسُّوا۟ بَأْسَنَآ إِذَا هُم مِّنْهَا يَرْكُضُونَ ‎</p>
 </td>
 <td width="50%">falam:a'ʔ 'ʔaḥas:uw' ba'ʔsana'ʔ 'ʔiða' hum m:inha' yarkuḍuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_013">13</a> ‏لَا تَرْكُضُوا۟ وَٱرْجِعُوٓا۟ إِلَىٰ مَآ أُتْرِفْتُمْ فِيهِ وَمَسَكِنِكُمْ لَعَلَّكُمْ تُسْـَٔلُونَ ‎</p>
 </td>
 <td width="50%">la' tarkuḍuw' wa'rjiʕuwʔ' 'ʔilay' ma'ʔ 'ʔutriftum fiyhi wamasakinikum laʕal:akum tusʔaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_014">14</a> ‏قَالُوا۟ يَوَيْلَنَآ إِنَّا كُنَّا ظَلِمِينَ ‎</p>
 </td>
 <td width="50%">qa'luw' yawaylana'ʔ 'ʔin:a' kun:a' ẓalimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_015">15</a> ‏فَمَا زَالَت تِّلْكَ دَعْوَىٰهُمْ حَتَّىٰ جَعَلْنَهُمْ حَصِيدًا خَمِدِينَ ‎</p>
 </td>
 <td width="50%">fama' za'lat t:ilka daʕway'hum ḥat:ay' jaʕalnahum ḥaṣiydan' xamidiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_016">16</a> ‏وَمَا خَلَقْنَا ٱلسَّمَآءَ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَعِبِينَ ‎</p>
 </td>
 <td width="50%">wama' xalaqna' 'ls:ama'ʔʔa wa'l'ʔarḍa wama' baynahuma' laʕibiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_017">17</a> ‏لَوْ أَرَدْنَآ أَن نَّتَّخِذَ لَهْوًۭا لَّٱتَّخَذْنَهُ مِن لَّدُنَّآ إِن كُنَّا فَعِلِينَ ‎</p>
 </td>
 <td width="50%">law 'ʔaradna'ʔ 'ʔan n:at:axiða lahwanm' l:a't:axaðnahu min l:adun:a'ʔ 'ʔin kun:a' faʕiliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_018">18</a> ‏بَلْ نَقْذِفُ بِٱلْحَقِّ عَلَى ٱلْبَطِلِ فَيَدْمَغُهُۥ فَإِذَا هُوَ زَاهِقٌۭ ۚ وَلَكُمُ ٱلْوَيْلُ مِمَّا تَصِفُونَ ‎</p>
 </td>
 <td width="50%">bal naqðifu bi'lḥaq:i ʕalay 'lbaṭili fayadmaɣuhuw fa'ʔiða' huwa za'hiqunm j walakumu 'lwaylu mim:a' taṣifuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_019">19</a> ‏وَلَهُۥ مَن فِى ٱلسَّمَوَتِ وَٱلْأَرْضِ ۚ وَمَنْ عِندَهُۥ لَا يَسْتَكْبِرُونَ عَنْ عِبَادَتِهِۦ وَلَا يَسْتَحْسِرُونَ ‎</p>
 </td>
 <td width="50%">walahuw man fiy 'ls:amawati wa'l'ʔarḍi j waman ʕindahuw la' yastakbiruwna ʕan ʕiba'datihiy wala' yastaḥsiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_020">20</a> ‏يُسَبِّحُونَ ٱلَّيْلَ وَٱلنَّهَارَ لَا يَفْتُرُونَ ‎</p>
 </td>
 <td width="50%">yusab:iḥuwna 'l:ayla wa'ln:aha'ra la' yafturuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_021">21</a> ‏أَمِ ٱتَّخَذُوٓا۟ ءَالِهَةًۭ مِّنَ ٱلْأَرْضِ هُمْ يُنشِرُونَ ‎</p>
 </td>
 <td width="50%">'ʔami 't:axaðuwʔ' ʔa'lihatanm m:ina 'l'ʔarḍi hum yunʃiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_022">22</a> ‏لَوْ كَانَ فِيهِمَآ ءَالِهَةٌ إِلَّا ٱللَّهُ لَفَسَدَتَا ۚ فَسُبْحَنَ ٱللَّهِ رَبِّ ٱلْعَرْشِ عَمَّا يَصِفُونَ ‎</p>
 </td>
 <td width="50%">law ka'na fiyhima'ʔ ʔa'lihatun 'ʔil:a' 'll:ahu lafasadata' j fasubḥana 'll:ahi rab:i 'lʕarʃi ʕam:a' yaṣifuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_023">23</a> ‏لَا يُسْـَٔلُ عَمَّا يَفْعَلُ وَهُمْ يُسْـَٔلُونَ ‎</p>
 </td>
 <td width="50%">la' yusʔalu ʕam:a' yafʕalu wahum yusʔaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_024">24</a> ‏أَمِ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ ءَالِهَةًۭ ۖ قُلْ هَاتُوا۟ بُرْهَنَكُمْ ۖ هَذَا ذِكْرُ مَن مَّعِىَ وَذِكْرُ مَن قَبْلِى ۗ بَلْ أَكْثَرُهُمْ لَا يَعْلَمُونَ ٱلْحَقَّ ۖ فَهُم مُّعْرِضُونَ ‎</p>
 </td>
 <td width="50%">'ʔami 't:axaðuw' min duwnihiyʔ ʔa'lihatanm sl' qul ha'tuw' burhanakum sl' haða' ðikru man m:aʕiya waðikru man qabliy ql' bal 'ʔakθaruhum la' yaʕlamuwna 'lḥaq:a sl' fahum m:uʕriḍuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_025">25</a> ‏وَمَآ أَرْسَلْنَا مِن قَبْلِكَ مِن رَّسُولٍ إِلَّا نُوحِىٓ إِلَيْهِ أَنَّهُۥ لَآ إِلَهَ إِلَّآ أَنَا۠ فَٱعْبُدُونِ ‎</p>
 </td>
 <td width="50%">wama'ʔ 'ʔarsalna' min qablika min r:asuwlin 'ʔil:a' nuwḥiyʔ 'ʔilayhi 'ʔan:ahuw la'ʔ 'ʔilaha 'ʔil:a'ʔ 'ʔana' fa'ʕbuduwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_026">26</a> ‏وَقَالُوا۟ ٱتَّخَذَ ٱلرَّحْمَنُ وَلَدًۭا ۗ سُبْحَنَهُۥ ۚ بَلْ عِبَادٌۭ مُّكْرَمُونَ ‎</p>
 </td>
 <td width="50%">waqa'luw' 't:axaða 'lr:aḥmanu waladanm' ql' subḥanahuw j bal ʕiba'dunm m:ukramuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_027">27</a> ‏لَا يَسْبِقُونَهُۥ بِٱلْقَوْلِ وَهُم بِأَمْرِهِۦ يَعْمَلُونَ ‎</p>
 </td>
 <td width="50%">la' yasbiquwnahuw bi'lqawli wahum bi'ʔamrihiy yaʕmaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_028">28</a> ‏يَعْلَمُ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْ وَلَا يَشْفَعُونَ إِلَّا لِمَنِ ٱرْتَضَىٰ وَهُم مِّنْ خَشْيَتِهِۦ مُشْفِقُونَ ‎</p>
 </td>
 <td width="50%">yaʕlamu ma' bayna 'ʔaydiyhim wama' xalfahum wala' yaʃfaʕuwna 'ʔil:a' limani 'rtaḍay' wahum m:in xaʃyatihiy muʃfiquwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_029">29</a> ‏ وَمَن يَقُلْ مِنْهُمْ إِنِّىٓ إِلَهٌۭ مِّن دُونِهِۦ فَذَلِكَ نَجْزِيهِ جَهَنَّمَ ۚ كَذَلِكَ نَجْزِى ٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%"> waman yaqul minhum 'ʔin:iyʔ 'ʔilahunm m:in duwnihiy faðalika najziyhi jahan:ama j kaðalika najziy 'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_030">30</a> ‏أَوَلَمْ يَرَ ٱلَّذِينَ كَفَرُوٓا۟ أَنَّ ٱلسَّمَوَتِ وَٱلْأَرْضَ كَانَتَا رَتْقًۭا فَفَتَقْنَهُمَا ۖ وَجَعَلْنَا مِنَ ٱلْمَآءِ كُلَّ شَىْءٍ حَىٍّ ۖ أَفَلَا يُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">'ʔawalam yara 'l:aðiyna kafaruwʔ' 'ʔan:a 'ls:amawati wa'l'ʔarḍa ka'nata' ratqanm' fafataqnahuma' sl' wajaʕalna' mina 'lma'ʔʔi kul:a ʃayʔin ḥay:in sl' 'ʔafala' yuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_031">31</a> ‏وَجَعَلْنَا فِى ٱلْأَرْضِ رَوَسِىَ أَن تَمِيدَ بِهِمْ وَجَعَلْنَا فِيهَا فِجَاجًۭا سُبُلًۭا لَّعَلَّهُمْ يَهْتَدُونَ ‎</p>
 </td>
 <td width="50%">wajaʕalna' fiy 'l'ʔarḍi rawasiya 'ʔan tamiyda bihim wajaʕalna' fiyha' fija'janm' subulanm' l:aʕal:ahum yahtaduwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_032">32</a> ‏وَجَعَلْنَا ٱلسَّمَآءَ سَقْفًۭا مَّحْفُوظًۭا ۖ وَهُمْ عَنْ ءَايَتِهَا مُعْرِضُونَ ‎</p>
 </td>
 <td width="50%">wajaʕalna' 'ls:ama'ʔʔa saqfanm' m:aḥfuwẓanm' sl' wahum ʕan ʔa'yatiha' muʕriḍuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_021_033">33</a> ‏وَهُوَ ٱلَّذِى خَلَقَ ٱلَّيْلَ وَٱلنَّهَارَ وَٱلشَّمْسَ وَٱلْقَمَرَ ۖ كُلٌّۭ فِى فَلَكٍۢ يَسْبَحُونَ ‎</p>
 </td>
 <td width="50%">wahuwa 'l:aðiy xalaqa 'l:ayla wa'ln:aha'ra wa'lʃ:amsa wa'lqamara sl' kul:unm fiy falakinm yasbaḥuwna </td>
