 <h1 align="CENTER">67. ‏الملك‎</h1>
 <h2 align="CENTER">'lmlk</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_001">1</a> ‏تَبَرَكَ ٱلَّذِى بِيَدِهِ ٱلْمُلْكُ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ ‎</p>
 </td>
 <td width="50%">tabaraka 'l:aðiy biyadihi 'lmulku wahuwa ʕalay' kul:i ʃayʔinm qadiyrun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_002">2</a> ‏ٱلَّذِى خَلَقَ ٱلْمَوْتَ وَٱلْحَيَوٰةَ لِيَبْلُوَكُمْ أَيُّكُمْ أَحْسَنُ عَمَلًۭا ۚ وَهُوَ ٱلْعَزِيزُ ٱلْغَفُورُ ‎</p>
 </td>
 <td width="50%">'l:aðiy xalaqa 'lmawta wa'lḥayaw'ta liyabluwakum 'ʔay:ukum 'ʔaḥsanu ʕamalanm' j wahuwa 'lʕaziyzu 'lɣafuwru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_003">3</a> ‏ٱلَّذِى خَلَقَ سَبْعَ سَمَوَتٍۢ طِبَاقًۭا ۖ مَّا تَرَىٰ فِى خَلْقِ ٱلرَّحْمَنِ مِن تَفَوُتٍۢ ۖ فَٱرْجِعِ ٱلْبَصَرَ هَلْ تَرَىٰ مِن فُطُورٍۢ ‎</p>
 </td>
 <td width="50%">'l:aðiy xalaqa sabʕa samawatinm ṭiba'qanm' sl' m:a' taray' fiy xalqi 'lr:aḥmani min tafawutinm sl' fa'rjiʕi 'lbaṣara hal taray' min fuṭuwrinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_004">4</a> ‏ثُمَّ ٱرْجِعِ ٱلْبَصَرَ كَرَّتَيْنِ يَنقَلِبْ إِلَيْكَ ٱلْبَصَرُ خَاسِئًۭا وَهُوَ حَسِيرٌۭ ‎</p>
 </td>
 <td width="50%">θum:a 'rjiʕi 'lbaṣara kar:atayni yanqalib 'ʔilayka 'lbaṣaru xa'siyʔanm' wahuwa ḥasiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_005">5</a> ‏وَلَقَدْ زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِمَصَبِيحَ وَجَعَلْنَهَا رُجُومًۭا لِّلشَّيَطِينِ ۖ وَأَعْتَدْنَا لَهُمْ عَذَابَ ٱلسَّعِيرِ ‎</p>
 </td>
 <td width="50%">walaqad zay:an:a' 'ls:ama'ʔʔa 'ld:unya' bimaṣabiyḥa wajaʕalnaha' rujuwmanm' l:ilʃ:ayaṭiyni sl' wa'ʔaʕtadna' lahum ʕaða'ba 'ls:aʕiyri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_006">6</a> ‏وَلِلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ عَذَابُ جَهَنَّمَ ۖ وَبِئْسَ ٱلْمَصِيرُ ‎</p>
 </td>
 <td width="50%">walil:aðiyna kafaruw' birab:ihim ʕaða'bu jahan:ama sl' wabiyʔsa 'lmaṣiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_007">7</a> ‏إِذَآ أُلْقُوا۟ فِيهَا سَمِعُوا۟ لَهَا شَهِيقًۭا وَهِىَ تَفُورُ ‎</p>
 </td>
 <td width="50%">'ʔiða'ʔ 'ʔulquw' fiyha' samiʕuw' laha' ʃahiyqanm' wahiya tafuwru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_008">8</a> ‏تَكَادُ تَمَيَّزُ مِنَ ٱلْغَيْظِ ۖ كُلَّمَآ أُلْقِىَ فِيهَا فَوْجٌۭ سَأَلَهُمْ خَزَنَتُهَآ أَلَمْ يَأْتِكُمْ نَذِيرٌۭ ‎</p>
 </td>
 <td width="50%">taka'du tamay:azu mina 'lɣayẓi sl' kul:ama'ʔ 'ʔulqiya fiyha' fawjunm sa'ʔalahum xazanatuha'ʔ 'ʔalam ya'ʔtikum naðiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_009">9</a> ‏قَالُوا۟ بَلَىٰ قَدْ جَآءَنَا نَذِيرٌۭ فَكَذَّبْنَا وَقُلْنَا مَا نَزَّلَ ٱللَّهُ مِن شَىْءٍ إِنْ أَنتُمْ إِلَّا فِى ضَلَلٍۢ كَبِيرٍۢ ‎</p>
 </td>
 <td width="50%">qa'luw' balay' qad ja'ʔʔana' naðiyrunm fakað:abna' waqulna' ma' naz:ala 'll:ahu min ʃayʔin 'ʔin 'ʔantum 'ʔil:a' fiy ḍalalinm kabiyrinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_010">10</a> ‏وَقَالُوا۟ لَوْ كُنَّا نَسْمَعُ أَوْ نَعْقِلُ مَا كُنَّا فِىٓ أَصْحَبِ ٱلسَّعِيرِ ‎</p>
 </td>
 <td width="50%">waqa'luw' law kun:a' nasmaʕu 'ʔaw naʕqilu ma' kun:a' fiyʔ 'ʔaṣḥabi 'ls:aʕiyri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_011">11</a> ‏فَٱعْتَرَفُوا۟ بِذَنۢبِهِمْ فَسُحْقًۭا لِأَصْحَبِ ٱلسَّعِيرِ ‎</p>
 </td>
 <td width="50%">fa'ʕtarafuw' biðanmbihim fasuḥqanm' li'ʔaṣḥabi 'ls:aʕiyri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_012">12</a> ‏إِنَّ ٱلَّذِينَ يَخْشَوْنَ رَبَّهُم بِٱلْغَيْبِ لَهُم مَّغْفِرَةٌۭ وَأَجْرٌۭ كَبِيرٌۭ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l:aðiyna yaxʃawna rab:ahum bi'lɣaybi lahum m:aɣfiratunm wa'ʔajrunm kabiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_013">13</a> ‏وَأَسِرُّوا۟ قَوْلَكُمْ أَوِ ٱجْهَرُوا۟ بِهِۦٓ ۖ إِنَّهُۥ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ ‎</p>
 </td>
 <td width="50%">wa'ʔasir:uw' qawlakum 'ʔawi 'jharuw' bihiyʔ sl' 'ʔin:ahuw ʕaliymunm biða'ti 'lṣ:uduwri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_014">14</a> ‏أَلَا يَعْلَمُ مَنْ خَلَقَ وَهُوَ ٱللَّطِيفُ ٱلْخَبِيرُ ‎</p>
 </td>
 <td width="50%">'ʔala' yaʕlamu man xalaqa wahuwa 'll:aṭiyfu 'lxabiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_015">15</a> ‏هُوَ ٱلَّذِى جَعَلَ لَكُمُ ٱلْأَرْضَ ذَلُولًۭا فَٱمْشُوا۟ فِى مَنَاكِبِهَا وَكُلُوا۟ مِن رِّزْقِهِۦ ۖ وَإِلَيْهِ ٱلنُّشُورُ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiy jaʕala lakumu 'l'ʔarḍa ðaluwlanm' fa'mʃuw' fiy mana'kibiha' wakuluw' min r:izqihiy sl' wa'ʔilayhi 'ln:uʃuwru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_016">16</a> ‏ءَأَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يَخْسِفَ بِكُمُ ٱلْأَرْضَ فَإِذَا هِىَ تَمُورُ ‎</p>
 </td>
 <td width="50%">ʔa'ʔamintum m:an fiy 'ls:ama'ʔʔi 'ʔan yaxsifa bikumu 'l'ʔarḍa fa'ʔiða' hiya tamuwru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_017">17</a> ‏أَمْ أَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يُرْسِلَ عَلَيْكُمْ حَاصِبًۭا ۖ فَسَتَعْلَمُونَ كَيْفَ نَذِيرِ ‎</p>
 </td>
 <td width="50%">'ʔam 'ʔamintum m:an fiy 'ls:ama'ʔʔi 'ʔan yursila ʕalaykum ḥa'ṣibanm' sl' fasataʕlamuwna kayfa naðiyri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_018">18</a> ‏وَلَقَدْ كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ فَكَيْفَ كَانَ نَكِيرِ ‎</p>
 </td>
 <td width="50%">walaqad kað:aba 'l:aðiyna min qablihim fakayfa ka'na nakiyri </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_019">19</a> ‏أَوَلَمْ يَرَوْا۟ إِلَى ٱلطَّيْرِ فَوْقَهُمْ صَٓفَّتٍۢ وَيَقْبِضْنَ ۚ مَا يُمْسِكُهُنَّ إِلَّا ٱلرَّحْمَنُ ۚ إِنَّهُۥ بِكُلِّ شَىْءٍۭ بَصِيرٌ ‎</p>
 </td>
 <td width="50%">'ʔawalam yaraw' 'ʔilay 'lṭ:ayri fawqahum ṣaʔf:atinm wayaqbiḍna j ma' yumsikuhun:a 'ʔil:a' 'lr:aḥmanu j 'ʔin:ahuw bikul:i ʃayʔinm baṣiyrun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_020">20</a> ‏أَمَّنْ هَذَا ٱلَّذِى هُوَ جُندٌۭ لَّكُمْ يَنصُرُكُم مِّن دُونِ ٱلرَّحْمَنِ ۚ إِنِ ٱلْكَفِرُونَ إِلَّا فِى غُرُورٍ ‎</p>
 </td>
 <td width="50%">'ʔam:an haða' 'l:aðiy huwa jundunm l:akum yanṣurukum m:in duwni 'lr:aḥmani j 'ʔini 'lkafiruwna 'ʔil:a' fiy ɣuruwrin </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_021">21</a> ‏أَمَّنْ هَذَا ٱلَّذِى يَرْزُقُكُمْ إِنْ أَمْسَكَ رِزْقَهُۥ ۚ بَل لَّجُّوا۟ فِى عُتُوٍّۢ وَنُفُورٍ ‎</p>
 </td>
 <td width="50%">'ʔam:an haða' 'l:aðiy yarzuqukum 'ʔin 'ʔamsaka rizqahuw j bal l:aj:uw' fiy ʕutuw:inm wanufuwrin </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_022">22</a> ‏أَفَمَن يَمْشِى مُكِبًّا عَلَىٰ وَجْهِهِۦٓ أَهْدَىٰٓ أَمَّن يَمْشِى سَوِيًّا عَلَىٰ صِرَطٍۢ مُّسْتَقِيمٍۢ ‎</p>
 </td>
 <td width="50%">'ʔafaman yamʃiy mukib:an' ʕalay' wajhihiyʔ 'ʔahday'ʔ 'ʔam:an yamʃiy sawiy:an' ʕalay' ṣiraṭinm m:ustaqiyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_023">23</a> ‏قُلْ هُوَ ٱلَّذِىٓ أَنشَأَكُمْ وَجَعَلَ لَكُمُ ٱلسَّمْعَ وَٱلْأَبْصَرَ وَٱلْأَفْـِٔدَةَ ۖ قَلِيلًۭا مَّا تَشْكُرُونَ ‎</p>
 </td>
 <td width="50%">qul huwa 'l:aðiyʔ 'ʔanʃa'ʔakum wajaʕala lakumu 'ls:amʕa wa'l'ʔabṣara wa'l'ʔafʔidata sl' qaliylanm' m:a' taʃkuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_024">24</a> ‏قُلْ هُوَ ٱلَّذِى ذَرَأَكُمْ فِى ٱلْأَرْضِ وَإِلَيْهِ تُحْشَرُونَ ‎</p>
 </td>
 <td width="50%">qul huwa 'l:aðiy ðara'ʔakum fiy 'l'ʔarḍi wa'ʔilayhi tuḥʃaruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_025">25</a> ‏وَيَقُولُونَ مَتَىٰ هَذَا ٱلْوَعْدُ إِن كُنتُمْ صَدِقِينَ ‎</p>
 </td>
 <td width="50%">wayaquwluwna matay' haða' 'lwaʕdu 'ʔin kuntum ṣadiqiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_026">26</a> ‏قُلْ إِنَّمَا ٱلْعِلْمُ عِندَ ٱللَّهِ وَإِنَّمَآ أَنَا۠ نَذِيرٌۭ مُّبِينٌۭ ‎</p>
 </td>
 <td width="50%">qul 'ʔin:ama' 'lʕilmu ʕinda 'll:ahi wa'ʔin:ama'ʔ 'ʔana' naðiyrunm m:ubiynunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_027">27</a> ‏فَلَمَّا رَأَوْهُ زُلْفَةًۭ سِيٓـَٔتْ وُجُوهُ ٱلَّذِينَ كَفَرُوا۟ وَقِيلَ هَذَا ٱلَّذِى كُنتُم بِهِۦ تَدَّعُونَ ‎</p>
 </td>
 <td width="50%">falam:a' ra'ʔawhu zulfatanm siyʔʔat wujuwhu 'l:aðiyna kafaruw' waqiyla haða' 'l:aðiy kuntum bihiy tad:aʕuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_028">28</a> ‏قُلْ أَرَءَيْتُمْ إِنْ أَهْلَكَنِىَ ٱللَّهُ وَمَن مَّعِىَ أَوْ رَحِمَنَا فَمَن يُجِيرُ ٱلْكَفِرِينَ مِنْ عَذَابٍ أَلِيمٍۢ ‎</p>
 </td>
 <td width="50%">qul 'ʔaraʔaytum 'ʔin 'ʔahlakaniya 'll:ahu waman m:aʕiya 'ʔaw raḥimana' faman yujiyru 'lkafiriyna min ʕaða'bin 'ʔaliyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_029">29</a> ‏قُلْ هُوَ ٱلرَّحْمَنُ ءَامَنَّا بِهِۦ وَعَلَيْهِ تَوَكَّلْنَا ۖ فَسَتَعْلَمُونَ مَنْ هُوَ فِى ضَلَلٍۢ مُّبِينٍۢ ‎</p>
 </td>
 <td width="50%">qul huwa 'lr:aḥmanu ʔa'man:a' bihiy waʕalayhi tawak:alna' sl' fasataʕlamuwna man huwa fiy ḍalalinm m:ubiyninm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_067_030">30</a> ‏قُلْ أَرَءَيْتُمْ إِنْ أَصْبَحَ مَآؤُكُمْ غَوْرًۭا فَمَن يَأْتِيكُم بِمَآءٍۢ مَّعِينٍۭ ‎</p>
 </td>
 <td width="50%">qul 'ʔaraʔaytum 'ʔin 'ʔaṣbaḥa ma'ʔwʔukum ɣawranm' faman ya'ʔtiykum bima'ʔʔinm m:aʕiyninm </td>
 </tr>
