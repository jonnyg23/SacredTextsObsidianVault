 <h1 align="CENTER">61. ‏الصف‎</h1>
 <h2 align="CENTER">'lṣf</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_001">1</a> ‏سَبَّحَ لِلَّهِ مَا فِى ٱلسَّمَوَتِ وَمَا فِى ٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">sab:aḥa lil:ahi ma' fiy 'ls:amawati wama' fiy 'l'ʔarḍi sl' wahuwa 'lʕaziyzu 'lḥakiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_002">2</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لِمَ تَقُولُونَ مَا لَا تَفْعَلُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' lima taquwluwna ma' la' tafʕaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_003">3</a> ‏كَبُرَ مَقْتًا عِندَ ٱللَّهِ أَن تَقُولُوا۟ مَا لَا تَفْعَلُونَ ‎</p>
 </td>
 <td width="50%">kabura maqtan' ʕinda 'll:ahi 'ʔan taquwluw' ma' la' tafʕaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_004">4</a> ‏إِنَّ ٱللَّهَ يُحِبُّ ٱلَّذِينَ يُقَتِلُونَ فِى سَبِيلِهِۦ صَفًّۭا كَأَنَّهُم بُنْيَنٌۭ مَّرْصُوصٌۭ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'll:aha yuḥib:u 'l:aðiyna yuqatiluwna fiy sabiylihiy ṣaf:anm' ka'ʔan:ahum bunyanunm m:arṣuwṣunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_005">5</a> ‏وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِۦ يَقَوْمِ لِمَ تُؤْذُونَنِى وَقَد تَّعْلَمُونَ أَنِّى رَسُولُ ٱللَّهِ إِلَيْكُمْ ۖ فَلَمَّا زَاغُوٓا۟ أَزَاغَ ٱللَّهُ قُلُوبَهُمْ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلْفَسِقِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔið qa'la muwsay' liqawmihiy yaqawmi lima tuwʔðuwnaniy waqad t:aʕlamuwna 'ʔan:iy rasuwlu 'll:ahi 'ʔilaykum sl' falam:a' za'ɣuwʔ' 'ʔaza'ɣa 'll:ahu quluwbahum j wa'll:ahu la' yahdiy 'lqawma 'lfasiqiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_006">6</a> ‏وَإِذْ قَالَ عِيسَى ٱبْنُ مَرْيَمَ يَبَنِىٓ إِسْرَٓءِيلَ إِنِّى رَسُولُ ٱللَّهِ إِلَيْكُم مُّصَدِّقًۭا لِّمَا بَيْنَ يَدَىَّ مِنَ ٱلتَّوْرَىٰةِ وَمُبَشِّرًۢا بِرَسُولٍۢ يَأْتِى مِنۢ بَعْدِى ٱسْمُهُۥٓ أَحْمَدُ ۖ فَلَمَّا جَآءَهُم بِٱلْبَيِّنَتِ قَالُوا۟ هَذَا سِحْرٌۭ مُّبِينٌۭ ‎</p>
 </td>
 <td width="50%">wa'ʔið qa'la ʕiysay 'bnu maryama yabaniyʔ 'ʔisraʔʔiyla 'ʔin:iy rasuwlu 'll:ahi 'ʔilaykum m:uṣad:iqanm' l:ima' bayna yaday:a mina 'lt:awray'ti wamubaʃ:iranm' birasuwlinm ya'ʔtiy minm baʕdiy 'smuhuwʔ 'ʔaḥmadu sl' falam:a' ja'ʔʔahum bi'lbay:inati qa'luw' haða' siḥrunm m:ubiynunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_007">7</a> ‏وَمَنْ أَظْلَمُ مِمَّنِ ٱفْتَرَىٰ عَلَى ٱللَّهِ ٱلْكَذِبَ وَهُوَ يُدْعَىٰٓ إِلَى ٱلْإِسْلَمِ ۚ وَٱللَّهُ لَا يَهْدِى ٱلْقَوْمَ ٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%">waman 'ʔaẓlamu mim:ani 'ftaray' ʕalay 'll:ahi 'lkaðiba wahuwa yudʕay'ʔ 'ʔilay 'l'ʔislami j wa'll:ahu la' yahdiy 'lqawma 'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_008">8</a> ‏يُرِيدُونَ لِيُطْفِـُٔوا۟ نُورَ ٱللَّهِ بِأَفْوَهِهِمْ وَٱللَّهُ مُتِمُّ نُورِهِۦ وَلَوْ كَرِهَ ٱلْكَفِرُونَ ‎</p>
 </td>
 <td width="50%">yuriyduwna liyuṭfiʔuw' nuwra 'll:ahi bi'ʔafwahihim wa'll:ahu mutim:u nuwrihiy walaw kariha 'lkafiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_009">9</a> ‏هُوَ ٱلَّذِىٓ أَرْسَلَ رَسُولَهُۥ بِٱلْهُدَىٰ وَدِينِ ٱلْحَقِّ لِيُظْهِرَهُۥ عَلَى ٱلدِّينِ كُلِّهِۦ وَلَوْ كَرِهَ ٱلْمُشْرِكُونَ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiyʔ 'ʔarsala rasuwlahuw bi'lhuday' wadiyni 'lḥaq:i liyuẓhirahuw ʕalay 'ld:iyni kul:ihiy walaw kariha 'lmuʃrikuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_010">10</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ هَلْ أَدُلُّكُمْ عَلَىٰ تِجَرَةٍۢ تُنجِيكُم مِّنْ عَذَابٍ أَلِيمٍۢ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' hal 'ʔadul:ukum ʕalay' tijaratinm tunjiykum m:in ʕaða'bin 'ʔaliyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_011">11</a> ‏تُؤْمِنُونَ بِٱللَّهِ وَرَسُولِهِۦ وَتُجَهِدُونَ فِى سَبِيلِ ٱللَّهِ بِأَمْوَلِكُمْ وَأَنفُسِكُمْ ۚ ذَلِكُمْ خَيْرٌۭ لَّكُمْ إِن كُنتُمْ تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">tuwʔminuwna bi'll:ahi warasuwlihiy watujahiduwna fiy sabiyli 'll:ahi bi'ʔamwalikum wa'ʔanfusikum j ðalikum xayrunm l:akum 'ʔin kuntum taʕlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_012">12</a> ‏يَغْفِرْ لَكُمْ ذُنُوبَكُمْ وَيُدْخِلْكُمْ جَنَّتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَرُ وَمَسَكِنَ طَيِّبَةًۭ فِى جَنَّتِ عَدْنٍۢ ۚ ذَلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ ‎</p>
 </td>
 <td width="50%">yaɣfir lakum ðunuwbakum wayudxilkum jan:atinm tajriy min taḥtiha' 'l'ʔanharu wamasakina ṭay:ibatanm fiy jan:ati ʕadninm j ðalika 'lfawzu 'lʕaẓiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_013">13</a> ‏وَأُخْرَىٰ تُحِبُّونَهَا ۖ نَصْرٌۭ مِّنَ ٱللَّهِ وَفَتْحٌۭ قَرِيبٌۭ ۗ وَبَشِّرِ ٱلْمُؤْمِنِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔuxray' tuḥib:uwnaha' sl' naṣrunm m:ina 'll:ahi wafatḥunm qariybunm ql' wabaʃ:iri 'lmuwʔminiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_061_014">14</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ كُونُوٓا۟ أَنصَارَ ٱللَّهِ كَمَا قَالَ عِيسَى ٱبْنُ مَرْيَمَ لِلْحَوَارِيِّۦنَ مَنْ أَنصَارِىٓ إِلَى ٱللَّهِ ۖ قَالَ ٱلْحَوَارِيُّونَ نَحْنُ أَنصَارُ ٱللَّهِ ۖ فَـَٔامَنَت طَّآئِفَةٌۭ مِّنۢ بَنِىٓ إِسْرَٓءِيلَ وَكَفَرَت طَّآئِفَةٌۭ ۖ فَأَيَّدْنَا ٱلَّذِينَ ءَامَنُوا۟ عَلَىٰ عَدُوِّهِمْ فَأَصْبَحُوا۟ ظَهِرِينَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' kuwnuwʔ' 'ʔanṣa'ra 'll:ahi kama' qa'la ʕiysay 'bnu maryama lilḥawa'riy:iyna man 'ʔanṣa'riyʔ 'ʔilay 'll:ahi sl' qa'la 'lḥawa'riy:uwna naḥnu 'ʔanṣa'ru 'll:ahi sl' faʔa'manat ṭ:a'ʔyʔifatunm m:inm baniyʔ 'ʔisraʔʔiyla wakafarat ṭ:a'ʔyʔifatunm sl' fa'ʔay:adna' 'l:aðiyna ʔa'manuw' ʕalay' ʕaduw:ihim fa'ʔaṣbaḥuw' ẓahiriyna </td>
 </tr>
 </table>
