 <h1 align="CENTER">16. ‏النحل‎</h1>
 <h2 align="CENTER">'lnḥl</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_001">1</a> ‏أَتَىٰٓ أَمْرُ ٱللَّهِ فَلَا تَسْتَعْجِلُوهُ ۚ سُبْحَنَهُۥ وَتَعَلَىٰ عَمَّا يُشْرِكُونَ ‎</p>
 </td>
 <td width="50%">'ʔatay'ʔ 'ʔamru 'll:ahi fala' tastaʕjiluwhu j subḥanahuw wataʕalay' ʕam:a' yuʃrikuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_002">2</a> ‏يُنَزِّلُ ٱلْمَلَٓئِكَةَ بِٱلرُّوحِ مِنْ أَمْرِهِۦ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦٓ أَنْ أَنذِرُوٓا۟ أَنَّهُۥ لَآ إِلَهَ إِلَّآ أَنَا۠ فَٱتَّقُونِ ‎</p>
 </td>
 <td width="50%">yunaz:ilu 'lmalaʔyʔikata bi'lr:uwḥi min 'ʔamrihiy ʕalay' man yaʃa'ʔʔu min ʕiba'dihiyʔ 'ʔan 'ʔanðiruwʔ' 'ʔan:ahuw la'ʔ 'ʔilaha 'ʔil:a'ʔ 'ʔana' fa't:aquwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_003">3</a> ‏خَلَقَ ٱلسَّمَوَتِ وَٱلْأَرْضَ بِٱلْحَقِّ ۚ تَعَلَىٰ عَمَّا يُشْرِكُونَ ‎</p>
 </td>
 <td width="50%">xalaqa 'ls:amawati wa'l'ʔarḍa bi'lḥaq:i j taʕalay' ʕam:a' yuʃrikuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_004">4</a> ‏خَلَقَ ٱلْإِنسَنَ مِن نُّطْفَةٍۢ فَإِذَا هُوَ خَصِيمٌۭ مُّبِينٌۭ ‎</p>
 </td>
 <td width="50%">xalaqa 'l'ʔinsana min n:uṭfatinm fa'ʔiða' huwa xaṣiymunm m:ubiynunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_005">5</a> ‏وَٱلْأَنْعَمَ خَلَقَهَا ۗ لَكُمْ فِيهَا دِفْءٌۭ وَمَنَفِعُ وَمِنْهَا تَأْكُلُونَ ‎</p>
 </td>
 <td width="50%">wa'l'ʔanʕama xalaqaha' ql' lakum fiyha' difʔunm wamanafiʕu waminha' ta'ʔkuluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_006">6</a> ‏وَلَكُمْ فِيهَا جَمَالٌ حِينَ تُرِيحُونَ وَحِينَ تَسْرَحُونَ ‎</p>
 </td>
 <td width="50%">walakum fiyha' jama'lun ḥiyna turiyḥuwna waḥiyna tasraḥuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_007">7</a> ‏وَتَحْمِلُ أَثْقَالَكُمْ إِلَىٰ بَلَدٍۢ لَّمْ تَكُونُوا۟ بَلِغِيهِ إِلَّا بِشِقِّ ٱلْأَنفُسِ ۚ إِنَّ رَبَّكُمْ لَرَءُوفٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">wataḥmilu 'ʔaθqa'lakum 'ʔilay' baladinm l:am takuwnuw' baliɣiyhi 'ʔil:a' biʃiq:i 'l'ʔanfusi j 'ʔin:a rab:akum laraʔuwfunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_008">8</a> ‏وَٱلْخَيْلَ وَٱلْبِغَالَ وَٱلْحَمِيرَ لِتَرْكَبُوهَا وَزِينَةًۭ ۚ وَيَخْلُقُ مَا لَا تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">wa'lxayla wa'lbiɣa'la wa'lḥamiyra litarkabuwha' waziynatanm j wayaxluqu ma' la' taʕlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_009">9</a> ‏وَعَلَى ٱللَّهِ قَصْدُ ٱلسَّبِيلِ وَمِنْهَا جَآئِرٌۭ ۚ وَلَوْ شَآءَ لَهَدَىٰكُمْ أَجْمَعِينَ ‎</p>
 </td>
 <td width="50%">waʕalay 'll:ahi qaṣdu 'ls:abiyli waminha' ja'ʔyʔirunm j walaw ʃa'ʔʔa lahaday'kum 'ʔajmaʕiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_010">10</a> ‏هُوَ ٱلَّذِىٓ أَنزَلَ مِنَ ٱلسَّمَآءِ مَآءًۭ ۖ لَّكُم مِّنْهُ شَرَابٌۭ وَمِنْهُ شَجَرٌۭ فِيهِ تُسِيمُونَ ‎</p>
 </td>
 <td width="50%">huwa 'l:aðiyʔ 'ʔanzala mina 'ls:ama'ʔʔi ma'ʔʔanm sl' l:akum m:inhu ʃara'bunm waminhu ʃajarunm fiyhi tusiymuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_011">11</a> ‏يُنۢبِتُ لَكُم بِهِ ٱلزَّرْعَ وَٱلزَّيْتُونَ وَٱلنَّخِيلَ وَٱلْأَعْنَبَ وَمِن كُلِّ ٱلثَّمَرَتِ ۗ إِنَّ فِى ذَلِكَ لَءَايَةًۭ لِّقَوْمٍۢ يَتَفَكَّرُونَ ‎</p>
 </td>
 <td width="50%">yunmbitu lakum bihi 'lz:arʕa wa'lz:aytuwna wa'ln:axiyla wa'l'ʔaʕnaba wamin kul:i 'lθ:amarati ql' 'ʔin:a fiy ðalika laʔa'yatanm l:iqawminm yatafak:aruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_012">12</a> ‏وَسَخَّرَ لَكُمُ ٱلَّيْلَ وَٱلنَّهَارَ وَٱلشَّمْسَ وَٱلْقَمَرَ ۖ وَٱلنُّجُومُ مُسَخَّرَتٌۢ بِأَمْرِهِۦٓ ۗ إِنَّ فِى ذَلِكَ لَءَايَتٍۢ لِّقَوْمٍۢ يَعْقِلُونَ ‎</p>
 </td>
 <td width="50%">wasax:ara lakumu 'l:ayla wa'ln:aha'ra wa'lʃ:amsa wa'lqamara sl' wa'ln:ujuwmu musax:aratunm bi'ʔamrihiyʔ ql' 'ʔin:a fiy ðalika laʔa'yatinm l:iqawminm yaʕqiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_013">13</a> ‏وَمَا ذَرَأَ لَكُمْ فِى ٱلْأَرْضِ مُخْتَلِفًا أَلْوَنُهُۥٓ ۗ إِنَّ فِى ذَلِكَ لَءَايَةًۭ لِّقَوْمٍۢ يَذَّكَّرُونَ ‎</p>
 </td>
 <td width="50%">wama' ðara'ʔa lakum fiy 'l'ʔarḍi muxtalifan' 'ʔalwanuhuwʔ ql' 'ʔin:a fiy ðalika laʔa'yatanm l:iqawminm yað:ak:aruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_014">14</a> ‏وَهُوَ ٱلَّذِى سَخَّرَ ٱلْبَحْرَ لِتَأْكُلُوا۟ مِنْهُ لَحْمًۭا طَرِيًّۭا وَتَسْتَخْرِجُوا۟ مِنْهُ حِلْيَةًۭ تَلْبَسُونَهَا وَتَرَى ٱلْفُلْكَ مَوَاخِرَ فِيهِ وَلِتَبْتَغُوا۟ مِن فَضْلِهِۦ وَلَعَلَّكُمْ تَشْكُرُونَ ‎</p>
 </td>
 <td width="50%">wahuwa 'l:aðiy sax:ara 'lbaḥra lita'ʔkuluw' minhu laḥmanm' ṭariy:anm' watastaxrijuw' minhu ḥilyatanm talbasuwnaha' wataray 'lfulka mawa'xira fiyhi walitabtaɣuw' min faḍlihiy walaʕal:akum taʃkuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_015">15</a> ‏وَأَلْقَىٰ فِى ٱلْأَرْضِ رَوَسِىَ أَن تَمِيدَ بِكُمْ وَأَنْهَرًۭا وَسُبُلًۭا لَّعَلَّكُمْ تَهْتَدُونَ ‎</p>
 </td>
 <td width="50%">wa'ʔalqay' fiy 'l'ʔarḍi rawasiya 'ʔan tamiyda bikum wa'ʔanharanm' wasubulanm' l:aʕal:akum tahtaduwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_016">16</a> ‏وَعَلَمَتٍۢ ۚ وَبِٱلنَّجْمِ هُمْ يَهْتَدُونَ ‎</p>
 </td>
 <td width="50%">waʕalamatinm j wabi'ln:ajmi hum yahtaduwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_017">17</a> ‏أَفَمَن يَخْلُقُ كَمَن لَّا يَخْلُقُ ۗ أَفَلَا تَذَكَّرُونَ ‎</p>
 </td>
 <td width="50%">'ʔafaman yaxluqu kaman l:a' yaxluqu ql' 'ʔafala' taðak:aruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_018">18</a> ‏وَإِن تَعُدُّوا۟ نِعْمَةَ ٱللَّهِ لَا تُحْصُوهَآ ۗ إِنَّ ٱللَّهَ لَغَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">wa'ʔin taʕud:uw' niʕmata 'll:ahi la' tuḥṣuwha'ʔ ql' 'ʔin:a 'll:aha laɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_019">19</a> ‏وَٱللَّهُ يَعْلَمُ مَا تُسِرُّونَ وَمَا تُعْلِنُونَ ‎</p>
 </td>
 <td width="50%">wa'll:ahu yaʕlamu ma' tusir:uwna wama' tuʕlinuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_020">20</a> ‏وَٱلَّذِينَ يَدْعُونَ مِن دُونِ ٱللَّهِ لَا يَخْلُقُونَ شَيْـًۭٔا وَهُمْ يُخْلَقُونَ ‎</p>
 </td>
 <td width="50%">wa'l:aðiyna yadʕuwna min duwni 'll:ahi la' yaxluquwna ʃayʔanm' wahum yuxlaquwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_021">21</a> ‏أَمْوَتٌ غَيْرُ أَحْيَآءٍۢ ۖ وَمَا يَشْعُرُونَ أَيَّانَ يُبْعَثُونَ ‎</p>
 </td>
 <td width="50%">'ʔamwatun ɣayru 'ʔaḥya'ʔʔinm sl' wama' yaʃʕuruwna 'ʔay:a'na yubʕaθuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_022">22</a> ‏إِلَهُكُمْ إِلَهٌۭ وَحِدٌۭ ۚ فَٱلَّذِينَ لَا يُؤْمِنُونَ بِٱلْءَاخِرَةِ قُلُوبُهُم مُّنكِرَةٌۭ وَهُم مُّسْتَكْبِرُونَ ‎</p>
 </td>
 <td width="50%">'ʔilahukum 'ʔilahunm waḥidunm j fa'l:aðiyna la' yuwʔminuwna bi'lʔa'xirati quluwbuhum m:unkiratunm wahum m:ustakbiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_023">23</a> ‏لَا جَرَمَ أَنَّ ٱللَّهَ يَعْلَمُ مَا يُسِرُّونَ وَمَا يُعْلِنُونَ ۚ إِنَّهُۥ لَا يُحِبُّ ٱلْمُسْتَكْبِرِينَ ‎</p>
 </td>
 <td width="50%">la' jarama 'ʔan:a 'll:aha yaʕlamu ma' yusir:uwna wama' yuʕlinuwna j 'ʔin:ahuw la' yuḥib:u 'lmustakbiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_024">24</a> ‏وَإِذَا قِيلَ لَهُم مَّاذَآ أَنزَلَ رَبُّكُمْ ۙ قَالُوٓا۟ أَسَطِيرُ ٱلْأَوَّلِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔiða' qiyla lahum m:a'ða'ʔ 'ʔanzala rab:ukum l' qa'luwʔ' 'ʔasaṭiyru 'l'ʔaw:aliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_025">25</a> ‏لِيَحْمِلُوٓا۟ أَوْزَارَهُمْ كَامِلَةًۭ يَوْمَ ٱلْقِيَمَةِ ۙ وَمِنْ أَوْزَارِ ٱلَّذِينَ يُضِلُّونَهُم بِغَيْرِ عِلْمٍ ۗ أَلَا سَآءَ مَا يَزِرُونَ ‎</p>
 </td>
 <td width="50%">liyaḥmiluwʔ' 'ʔawza'rahum ka'milatanm yawma 'lqiyamati l' wamin 'ʔawza'ri 'l:aðiyna yuḍil:uwnahum biɣayri ʕilmin ql' 'ʔala' sa'ʔʔa ma' yaziruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_026">26</a> ‏قَدْ مَكَرَ ٱلَّذِينَ مِن قَبْلِهِمْ فَأَتَى ٱللَّهُ بُنْيَنَهُم مِّنَ ٱلْقَوَاعِدِ فَخَرَّ عَلَيْهِمُ ٱلسَّقْفُ مِن فَوْقِهِمْ وَأَتَىٰهُمُ ٱلْعَذَابُ مِنْ حَيْثُ لَا يَشْعُرُونَ ‎</p>
 </td>
 <td width="50%">qad makara 'l:aðiyna min qablihim fa'ʔatay 'll:ahu bunyanahum m:ina 'lqawa'ʕidi faxar:a ʕalayhimu 'ls:aqfu min fawqihim wa'ʔatay'humu 'lʕaða'bu min ḥayθu la' yaʃʕuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_027">27</a> ‏ثُمَّ يَوْمَ ٱلْقِيَمَةِ يُخْزِيهِمْ وَيَقُولُ أَيْنَ شُرَكَآءِىَ ٱلَّذِينَ كُنتُمْ تُشَٓقُّونَ فِيهِمْ ۚ قَالَ ٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ إِنَّ ٱلْخِزْىَ ٱلْيَوْمَ وَٱلسُّوٓءَ عَلَى ٱلْكَفِرِينَ ‎</p>
 </td>
 <td width="50%">θum:a yawma 'lqiyamati yuxziyhim wayaquwlu 'ʔayna ʃuraka'ʔʔiya 'l:aðiyna kuntum tuʃaʔq:uwna fiyhim j qa'la 'l:aðiyna 'ʔuwtuw' 'lʕilma 'ʔin:a 'lxizya 'lyawma wa'ls:uwʔʔa ʕalay 'lkafiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_028">28</a> ‏ٱلَّذِينَ تَتَوَفَّىٰهُمُ ٱلْمَلَٓئِكَةُ ظَالِمِىٓ أَنفُسِهِمْ ۖ فَأَلْقَوُا۟ ٱلسَّلَمَ مَا كُنَّا نَعْمَلُ مِن سُوٓءٍۭ ۚ بَلَىٰٓ إِنَّ ٱللَّهَ عَلِيمٌۢ بِمَا كُنتُمْ تَعْمَلُونَ ‎</p>
 </td>
 <td width="50%">'l:aðiyna tatawaf:ay'humu 'lmalaʔyʔikatu ẓa'limiyʔ 'ʔanfusihim sl' fa'ʔalqawu' 'ls:alama ma' kun:a' naʕmalu min suwʔʔinm j balay'ʔ 'ʔin:a 'll:aha ʕaliymunm bima' kuntum taʕmaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_029">29</a> ‏فَٱدْخُلُوٓا۟ أَبْوَبَ جَهَنَّمَ خَلِدِينَ فِيهَا ۖ فَلَبِئْسَ مَثْوَى ٱلْمُتَكَبِّرِينَ ‎</p>
 </td>
 <td width="50%">fa'dxuluwʔ' 'ʔabwaba jahan:ama xalidiyna fiyha' sl' falabiyʔsa maθway 'lmutakab:iriyna </td>
 </tr>
 <tr valign="TOP">

 <td width="50%">man kafara bi'll:ahi minm baʕdi 'ʔiymanihiyʔ 'ʔil:a' man 'ʔukriha waqalbuhuw muṭmayʔin:unm bi'l'ʔiymani walakin m:an ʃaraḥa bi'lkufri ṣadranm' faʕalayhim ɣaḍabunm m:ina 'll:ahi walahum ʕaða'bun ʕaẓiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_107">107</a> ‏ذَلِكَ بِأَنَّهُمُ ٱسْتَحَبُّوا۟ ٱلْحَيَوٰةَ ٱلدُّنْيَا عَلَى ٱلْءَاخِرَةِ وَأَنَّ ٱللَّهَ لَا يَهْدِى ٱلْقَوْمَ ٱلْكَفِرِينَ ‎</p>
 </td>
 <td width="50%">ðalika bi'ʔan:ahumu 'staḥab:uw' 'lḥayaw'ta 'ld:unya' ʕalay 'lʔa'xirati wa'ʔan:a 'll:aha la' yahdiy 'lqawma 'lkafiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_108">108</a> ‏أُو۟لَٓئِكَ ٱلَّذِينَ طَبَعَ ٱللَّهُ عَلَىٰ قُلُوبِهِمْ وَسَمْعِهِمْ وَأَبْصَرِهِمْ ۖ وَأُو۟لَٓئِكَ هُمُ ٱلْغَفِلُونَ ‎</p>
 </td>
 <td width="50%">'ʔuwlaʔyʔika 'l:aðiyna ṭabaʕa 'll:ahu ʕalay' quluwbihim wasamʕihim wa'ʔabṣarihim sl' wa'ʔuwlaʔyʔika humu 'lɣafiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_109">109</a> ‏لَا جَرَمَ أَنَّهُمْ فِى ٱلْءَاخِرَةِ هُمُ ٱلْخَسِرُونَ ‎</p>
 </td>
 <td width="50%">la' jarama 'ʔan:ahum fiy 'lʔa'xirati humu 'lxasiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_110">110</a> ‏ثُمَّ إِنَّ رَبَّكَ لِلَّذِينَ هَاجَرُوا۟ مِنۢ بَعْدِ مَا فُتِنُوا۟ ثُمَّ جَهَدُوا۟ وَصَبَرُوٓا۟ إِنَّ رَبَّكَ مِنۢ بَعْدِهَا لَغَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">θum:a 'ʔin:a rab:aka lil:aðiyna ha'jaruw' minm baʕdi ma' futinuw' θum:a jahaduw' waṣabaruwʔ' 'ʔin:a rab:aka minm baʕdiha' laɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_111">111</a> ‏ يَوْمَ تَأْتِى كُلُّ نَفْسٍۢ تُجَدِلُ عَن نَّفْسِهَا وَتُوَفَّىٰ كُلُّ نَفْسٍۢ مَّا عَمِلَتْ وَهُمْ لَا يُظْلَمُونَ ‎</p>
 </td>
 <td width="50%"> yawma ta'ʔtiy kul:u nafsinm tujadilu ʕan n:afsiha' watuwaf:ay' kul:u nafsinm m:a' ʕamilat wahum la' yuẓlamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_112">112</a> ‏وَضَرَبَ ٱللَّهُ مَثَلًۭا قَرْيَةًۭ كَانَتْ ءَامِنَةًۭ مُّطْمَئِنَّةًۭ يَأْتِيهَا رِزْقُهَا رَغَدًۭا مِّن كُلِّ مَكَانٍۢ فَكَفَرَتْ بِأَنْعُمِ ٱللَّهِ فَأَذَقَهَا ٱللَّهُ لِبَاسَ ٱلْجُوعِ وَٱلْخَوْفِ بِمَا كَانُوا۟ يَصْنَعُونَ ‎</p>
 </td>
 <td width="50%">waḍaraba 'll:ahu maθalanm' qaryatanm ka'nat ʔa'minatanm m:uṭmayʔin:atanm ya'ʔtiyha' rizquha' raɣadanm' m:in kul:i maka'ninm fakafarat bi'ʔanʕumi 'll:ahi fa'ʔaðaqaha' 'll:ahu liba'sa 'ljuwʕi wa'lxawfi bima' ka'nuw' yaṣnaʕuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_113">113</a> ‏وَلَقَدْ جَآءَهُمْ رَسُولٌۭ مِّنْهُمْ فَكَذَّبُوهُ فَأَخَذَهُمُ ٱلْعَذَابُ وَهُمْ ظَلِمُونَ ‎</p>
 </td>
 <td width="50%">walaqad ja'ʔʔahum rasuwlunm m:inhum fakað:abuwhu fa'ʔaxaðahumu 'lʕaða'bu wahum ẓalimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_114">114</a> ‏فَكُلُوا۟ مِمَّا رَزَقَكُمُ ٱللَّهُ حَلَلًۭا طَيِّبًۭا وَٱشْكُرُوا۟ نِعْمَتَ ٱللَّهِ إِن كُنتُمْ إِيَّاهُ تَعْبُدُونَ ‎</p>
 </td>
 <td width="50%">fakuluw' mim:a' razaqakumu 'll:ahu ḥalalanm' ṭay:ibanm' wa'ʃkuruw' niʕmata 'll:ahi 'ʔin kuntum 'ʔiy:a'hu taʕbuduwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_115">115</a> ‏إِنَّمَا حَرَّمَ عَلَيْكُمُ ٱلْمَيْتَةَ وَٱلدَّمَ وَلَحْمَ ٱلْخِنزِيرِ وَمَآ أُهِلَّ لِغَيْرِ ٱللَّهِ بِهِۦ ۖ فَمَنِ ٱضْطُرَّ غَيْرَ بَاغٍۢ وَلَا عَادٍۢ فَإِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">'ʔin:ama' ḥar:ama ʕalaykumu 'lmaytata wa'ld:ama walaḥma 'lxinziyri wama'ʔ 'ʔuhil:a liɣayri 'll:ahi bihiy sl' famani 'ḍṭur:a ɣayra ba'ɣinm wala' ʕa'dinm fa'ʔin:a 'll:aha ɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_116">116</a> ‏وَلَا تَقُولُوا۟ لِمَا تَصِفُ أَلْسِنَتُكُمُ ٱلْكَذِبَ هَذَا حَلَلٌۭ وَهَذَا حَرَامٌۭ لِّتَفْتَرُوا۟ عَلَى ٱللَّهِ ٱلْكَذِبَ ۚ إِنَّ ٱلَّذِينَ يَفْتَرُونَ عَلَى ٱللَّهِ ٱلْكَذِبَ لَا يُفْلِحُونَ ‎</p>
 </td>
 <td width="50%">wala' taquwluw' lima' taṣifu 'ʔalsinatukumu 'lkaðiba haða' ḥalalunm wahaða' ḥara'munm l:itaftaruw' ʕalay 'll:ahi 'lkaðiba j 'ʔin:a 'l:aðiyna yaftaruwna ʕalay 'll:ahi 'lkaðiba la' yufliḥuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_117">117</a> ‏مَتَعٌۭ قَلِيلٌۭ وَلَهُمْ عَذَابٌ أَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">mataʕunm qaliylunm walahum ʕaða'bun 'ʔaliymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_118">118</a> ‏وَعَلَى ٱلَّذِينَ هَادُوا۟ حَرَّمْنَا مَا قَصَصْنَا عَلَيْكَ مِن قَبْلُ ۖ وَمَا ظَلَمْنَهُمْ وَلَكِن كَانُوٓا۟ أَنفُسَهُمْ يَظْلِمُونَ ‎</p>
 </td>
 <td width="50%">waʕalay 'l:aðiyna ha'duw' ḥar:amna' ma' qaṣaṣna' ʕalayka min qablu sl' wama' ẓalamnahum walakin ka'nuwʔ' 'ʔanfusahum yaẓlimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_119">119</a> ‏ثُمَّ إِنَّ رَبَّكَ لِلَّذِينَ عَمِلُوا۟ ٱلسُّوٓءَ بِجَهَلَةٍۢ ثُمَّ تَابُوا۟ مِنۢ بَعْدِ ذَلِكَ وَأَصْلَحُوٓا۟ إِنَّ رَبَّكَ مِنۢ بَعْدِهَا لَغَفُورٌۭ رَّحِيمٌ ‎</p>
 </td>
 <td width="50%">θum:a 'ʔin:a rab:aka lil:aðiyna ʕamiluw' 'ls:uwʔʔa bijahalatinm θum:a ta'buw' minm baʕdi ðalika wa'ʔaṣlaḥuwʔ' 'ʔin:a rab:aka minm baʕdiha' laɣafuwrunm r:aḥiymun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_120">120</a> ‏إِنَّ إِبْرَهِيمَ كَانَ أُمَّةًۭ قَانِتًۭا لِّلَّهِ حَنِيفًۭا وَلَمْ يَكُ مِنَ ٱلْمُشْرِكِينَ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'ʔibrahiyma ka'na 'ʔum:atanm qa'nitanm' l:il:ahi ḥaniyfanm' walam yaku mina 'lmuʃrikiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_121">121</a> ‏شَاكِرًۭا لِأَنْعُمِهِ ۚ ٱجْتَبَىٰهُ وَهَدَىٰهُ إِلَىٰ صِرَطٍۢ مُّسْتَقِيمٍۢ ‎</p>
 </td>
 <td width="50%">ʃa'kiranm' li'ʔanʕumihi j 'jtabay'hu wahaday'hu 'ʔilay' ṣiraṭinm m:ustaqiyminm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_122">122</a> ‏وَءَاتَيْنَهُ فِى ٱلدُّنْيَا حَسَنَةًۭ ۖ وَإِنَّهُۥ فِى ٱلْءَاخِرَةِ لَمِنَ ٱلصَّلِحِينَ ‎</p>
 </td>
 <td width="50%">waʔa'taynahu fiy 'ld:unya' ḥasanatanm sl' wa'ʔin:ahuw fiy 'lʔa'xirati lamina 'lṣ:aliḥiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_123">123</a> ‏ثُمَّ أَوْحَيْنَآ إِلَيْكَ أَنِ ٱتَّبِعْ مِلَّةَ إِبْرَهِيمَ حَنِيفًۭا ۖ وَمَا كَانَ مِنَ ٱلْمُشْرِكِينَ ‎</p>
 </td>
 <td width="50%">θum:a 'ʔawḥayna'ʔ 'ʔilayka 'ʔani 't:abiʕ mil:ata 'ʔibrahiyma ḥaniyfanm' sl' wama' ka'na mina 'lmuʃrikiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_124">124</a> ‏إِنَّمَا جُعِلَ ٱلسَّبْتُ عَلَى ٱلَّذِينَ ٱخْتَلَفُوا۟ فِيهِ ۚ وَإِنَّ رَبَّكَ لَيَحْكُمُ بَيْنَهُمْ يَوْمَ ٱلْقِيَمَةِ فِيمَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ ‎</p>
 </td>
 <td width="50%">'ʔin:ama' juʕila 'ls:abtu ʕalay 'l:aðiyna 'xtalafuw' fiyhi j wa'ʔin:a rab:aka layaḥkumu baynahum yawma 'lqiyamati fiyma' ka'nuw' fiyhi yaxtalifuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_125">125</a> ‏ٱدْعُ إِلَىٰ سَبِيلِ رَبِّكَ بِٱلْحِكْمَةِ وَٱلْمَوْعِظَةِ ٱلْحَسَنَةِ ۖ وَجَدِلْهُم بِٱلَّتِى هِىَ أَحْسَنُ ۚ إِنَّ رَبَّكَ هُوَ أَعْلَمُ بِمَن ضَلَّ عَن سَبِيلِهِۦ ۖ وَهُوَ أَعْلَمُ بِٱلْمُهْتَدِينَ ‎</p>
 </td>
 <td width="50%">'dʕu 'ʔilay' sabiyli rab:ika bi'lḥikmati wa'lmawʕiẓati 'lḥasanati sl' wajadilhum bi'l:atiy hiya 'ʔaḥsanu j 'ʔin:a rab:aka huwa 'ʔaʕlamu biman ḍal:a ʕan sabiylihiy sl' wahuwa 'ʔaʕlamu bi'lmuhtadiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_126">126</a> ‏وَإِنْ عَاقَبْتُمْ فَعَاقِبُوا۟ بِمِثْلِ مَا عُوقِبْتُم بِهِۦ ۖ وَلَىِٕن صَبَرْتُمْ لَهُوَ خَيْرٌۭ لِّلصَّبِرِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔin ʕa'qabtum faʕa'qibuw' bimiθli ma' ʕuwqibtum bihiy sl' walayʔin ṣabartum lahuwa xayrunm l:ilṣ:abiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_127">127</a> ‏وَٱصْبِرْ وَمَا صَبْرُكَ إِلَّا بِٱللَّهِ ۚ وَلَا تَحْزَنْ عَلَيْهِمْ وَلَا تَكُ فِى ضَيْقٍۢ مِّمَّا يَمْكُرُونَ ‎</p>
 </td>
 <td width="50%">wa'ṣbir wama' ṣabruka 'ʔil:a' bi'll:ahi j wala' taḥzan ʕalayhim wala' taku fiy ḍayqinm m:im:a' yamkuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_016_128">128</a> ‏إِنَّ ٱللَّهَ مَعَ ٱلَّذِينَ ٱتَّقَوا۟ وَّٱلَّذِينَ هُم مُّحْسِنُونَ ‎</p>
 </td>
