 <tr><td width="8%" valign="TOP">
 <p>a:</p></td>

 </tr>
 <tr><td width="8%" valign="TOP">
 <p>â</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>what</i>; Germ. <i>man</i></p></td>

 <tr><td width="8%" valign="TOP">
 <p>ä</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>hat</i>, <i>man</i>, <i>ran</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>ai</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>aisle</i>, as i in <i>mine</i>; <i>bind</i>; Germ. <i>Hain</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>au</p></td>

 <p>d</p></td>

 </tr>
 <tr><td width="8%" valign="TOP">
 <p>e</p></td>
 <td width="92%" valign="TOP">
 <p>as e in <i>they</i>, as a in may; Fr. <i>ne</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>ê</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>met</i>, <i>get</i>, <i>then</i>; Germ. <i>denn</i>; Fr. <i>sienne</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>g</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>gig</i>; Germ. <i>geben</i>; Fr. <i>gout</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>h</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>has</i>, <i>he</i>; Germ. <i>haben</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>i</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>pique</i>, <i>machine</i>; ie as ye in English <i>yea</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>i:</p></td>
 <td width="92%" valign="TOP">

 </tr>
 <tr><td width="8%" valign="TOP">
 <p>î</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>pick</i>, <i>pit</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>j</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>judge</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>k</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>kick</i>, <i>kin</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>n</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>no</i>, <i>nun</i>, <i>not</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>ñ</p></td>
 <td width="92%" valign="TOP">
 <p>as ng in <i>ring</i>, <i>sing</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>o</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>note</i>, <i>boat</i></p></td>

 <p>q</p></td>
 <td width="92%" valign="TOP">

 <tr><td width="8%" valign="TOP">
 <p>s</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>see</i>, <i>sat</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>t</p></td>
 <td width="92%" valign="TOP">
 <p>pronounced with the tip of the tongue on the upper teeth</p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>u</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>rule</i>; Germ. <i>du</i>; Fr. on in <i>doux</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>û</p></td>

 <p>w</p></td>

 <p>as in <i>wit</i>, <i>win</i></p></td>

 <tr><td width="8%" valign="TOP">
 <p>y</p></td>
 <td width="92%" valign="TOP">
 <p>as in <i>yes</i>, <i>yet</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>dj</p></td>
 <td width="92%" valign="TOP">
 <p>as j in <i>judge</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>tc</p></td>
 <td width="92%" valign="TOP">
 <p>as ch in <i>church</i>; tci-chee as in <i>cheese</i></p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <sup><p>n</p></sup></td>
 <td width="92%" valign="TOP">
 <p>marks nasalized vowels; as a<sup>n</sup>, e<sup>n</sup>, ê<sup>n</sup>, o<sup>n</sup>, â<sup>n</sup>, ai<sup>n</sup>, etc.</p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>`</p></td>
 <td width="92%" valign="TOP">
 <p>indicates an aspiration or soft emission of the breadth which is initial or final, thus `h, ê<sup>n`</sup>, o`, etc.</p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>?</p></td>
 <td width="92%" valign="TOP">
 <p>marks a sudden closure of the glottis preceding or following a sound, thus ?a, o?, i?, â?, etc.</p></td>
 </tr>
 <tr><td width="8%" valign="TOP">
 <p>'</p></td>
 <td width="92%" valign="TOP">
 <p>marks the accented syllable of a word</p></td>
 </tr>
 </table>
 
 <i></i><p>t and <i>h</i> in this system are always pronounced separately.</p>
